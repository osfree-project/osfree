63/*-----------------------------------------------------------------
64 * The standard io loop that keeps submitting an io until it's
65 * all gone.
66 *---------------------------------------------------------------*/
67static int _io(struct device_area *where, char *buffer, int should_write)
68{
69 int fd = dev_fd(where->dev);
70 ssize_t n = 0;
71 size_t total = 0;
72
73 if (fd < 0) {
74 log_error("Attempt to read an unopened device (%s).",
75 dev_name(where->dev));
76 return 0;
77 }
78
79 /*
80 * Skip all writes in test mode.
81 */
82 if (should_write && test_mode())
83 return 1;
84
85 if (where->size > SSIZE_MAX) {
86 log_error("Read size too large: %" PRIu64, where->size);
87 return 0;
88 }
89
90 if (lseek(fd, (off_t) where->start, SEEK_SET) < 0) {
91 log_error("%s: lseek %" PRIu64 " failed: %s",
92 dev_name(where->dev), (uint64_t) where->start,
93 strerror(errno));
94 return 0;
95 }
96
97 while (total < (size_t) where->size) {
98 do
99 n = should_write ?
100 write(fd, buffer, (size_t) where->size - total) :
101 read(fd, buffer, (size_t) where->size - total);
102 while ((n < 0) && ((errno == EINTR) || (errno == EAGAIN)));
103
104 if (n < 0)
105 log_error("%s: %s failed after %" PRIu64 " of %" PRIu64
106 " at %" PRIu64 ": %s", dev_name(where->dev),
107 should_write ? "write" : "read",
108 (uint64_t) total,
109 (uint64_t) where->size,
110 (uint64_t) where->start, strerror(errno));
111
112 if (n <= 0)
113 break;
114
115 total += n;
116 buffer += n;
117 }
118
119 return (total == (size_t) where->size);
120}

129/*
130 * Get the sector size from an _open_ device.

132static int _get_block_size(struct device *dev, unsigned int *size)
133{
134 const char *name = dev_name(dev);
135
136 if ((dev->block_size == -1)) {
137 if (ioctl(dev_fd(dev), BLKBSZGET, &dev->block_size) < 0) {
138 log_sys_error("ioctl BLKBSZGET", name);
139 return 0;
140 }
141 log_debug("%s: block size is %u bytes", name, dev->block_size);
142 }
143
144 *size = (unsigned int) dev->block_size;
145
146 return 1;
147}

149/*
150 * Widens a region to be an aligned region.
151 */
152static void _widen_region(unsigned int block_size, struct device_area *region,
153 struct device_area *result)
154{
155 uint64_t mask = block_size - 1, delta;
156 memcpy(result, region, sizeof(*result));
157
158 /* adjust the start */
159 delta = result->start & mask;
160 if (delta) {
161 result->start -= delta;
162 result->size += delta;
163 }
164
165 /* adjust the end */
166 delta = (result->start + result->size) & mask;
167 if (delta)
168 result->size += block_size - delta;
169}

171static int _aligned_io(struct device_area *where, char *buffer,
172 int should_write)
173{
174 void *bounce;
175 unsigned int block_size = 0;
176 uintptr_t mask;
177 struct device_area widened;
178
179 if (!(where->dev->flags & DEV_REGULAR) &&
180 !_get_block_size(where->dev, &block_size)) {
181 stack;
182 return 0;
183 }
184
185 if (!block_size)
186 block_size = lvm_getpagesize();
187
188 _widen_region(block_size, where, &widened);
189
190 /* Do we need to use a bounce buffer? */
191 mask = block_size - 1;
192 if (!memcmp(where, &widened, sizeof(widened)) &&
193 !((uintptr_t) buffer & mask))
194 return _io(where, buffer, should_write);
195
196 /* Allocate a bounce buffer with an extra block */
197 if (!(bounce = alloca((size_t) widened.size + block_size))) {
198 log_error("Bounce buffer alloca failed");
199 return 0;
200 }
201
202 /*
203 * Realign start of bounce buffer (using the extra sector)
204 */
205 if (((uintptr_t) bounce) & mask)
206 bounce = (void *) ((((uintptr_t) bounce) + mask) & ~mask);
207
208 /* channel the io through the bounce buffer */
209 if (!_io(&widened, bounce, 0)) {
210 if (!should_write) {
211 stack;
212 return 0;
213 }
214 /* FIXME pre-extend the file */
215 memset(bounce, '\n', widened.size);
216 }
217
218 if (should_write) {
219 memcpy(bounce + (where->start - widened.start), buffer,
220 (size_t) where->size);
221
222 /* ... then we write */
223 return _io(&widened, bounce, 1);
224 }
225
226 memcpy(buffer, bounce + (where->start - widened.start),
227 (size_t) where->size);
228
229 return 1;
230}

250static int _dev_get_size_dev(const struct device *dev, uint64_t *size)
251{
252 int fd;
253 const char *name = dev_name(dev);
254
255 if ((fd = open(name, O_RDONLY)) < 0) {
256 log_sys_error("open", name);
257 return 0;
258 }
259
260 if (ioctl(fd, BLKGETSIZE64, size) < 0) {
261 log_sys_error("ioctl BLKGETSIZE64", name);
262 if (close(fd))
263 log_sys_error("close", name);
264 return 0;
265 }
266
267 *size >>= BLKSIZE_SHIFT; /* Convert to sectors */
268 if (close(fd))
269 log_sys_error("close", name);
270
271 log_very_verbose("%s: size is %" PRIu64 " sectors", name, *size);
272
273 return 1;
274}

329int dev_open_flags(struct device *dev, int flags, int direct, int quiet)
330{
331 struct stat buf;
332 const char *name;
333 int need_excl = 0, need_rw = 0;
334
335 if ((flags & O_ACCMODE) == O_RDWR)
336 need_rw = 1;
337
338 if ((flags & O_EXCL))
339 need_excl = 1;
340
341 if (dev->fd >= 0) {
342 if (((dev->flags & DEV_OPENED_RW) || !need_rw) &&
343 ((dev->flags & DEV_OPENED_EXCL) || !need_excl)) {
344 dev->open_count++;
345 return 1;
346 }
347
348 if (dev->open_count && !need_excl) {
349 /* FIXME Ensure we never get here */
350 log_debug("WARNING: %s already opened read-only",
351 dev_name(dev));
352 dev->open_count++;
353 }
354
355 dev_close_immediate(dev);
356 }
357
358 if (memlock())
359 log_error("WARNING: dev_open(%s) called while suspended",
360 dev_name(dev));
361
362 if (dev->flags & DEV_REGULAR)
363 name = dev_name(dev);
364 else if (!(name = dev_name_confirmed(dev, quiet))) {
365 stack;
366 return 0;
367 }
368
369 if (!(dev->flags & DEV_REGULAR) &&
370 ((stat(name, &buf) < 0) || (buf.st_rdev != dev->dev))) {
371 log_error("%s: stat failed: Has device name changed?", name);
372 return 0;
373 }
374
375#ifdef O_DIRECT_SUPPORT
376 if (direct) {
377 if (!(dev->flags & DEV_O_DIRECT_TESTED))
378 dev->flags |= DEV_O_DIRECT;
379
380 if ((dev->flags & DEV_O_DIRECT))
381 flags |= O_DIRECT;
382 }
383#endif
384
385#ifdef O_NOATIME
386 /* Don't update atime on device inodes */
387 if (!(dev->flags & DEV_REGULAR))
388 flags |= O_NOATIME;
389#endif
390
391 if ((dev->fd = open(name, flags, 0777)) < 0) {
392#ifdef O_DIRECT_SUPPORT
393 if (direct && !(dev->flags & DEV_O_DIRECT_TESTED)) {
394 flags &= ~O_DIRECT;
395 if ((dev->fd = open(name, flags, 0777)) >= 0) {
396 dev->flags &= ~DEV_O_DIRECT;
397 log_debug("%s: Not using O_DIRECT", name);
398 goto opened;
399 }
400 }
401#endif
402 if (quiet)
403 log_sys_debug("open", name);
404 else
405 log_sys_error("open", name);
406 return 0;
407 }
408
409#ifdef O_DIRECT_SUPPORT
410 opened:
411 if (direct)
412 dev->flags |= DEV_O_DIRECT_TESTED;
413#endif
414 dev->open_count++;
415 dev->flags &= ~DEV_ACCESSED_W;
416
417 if (need_rw)
418 dev->flags |= DEV_OPENED_RW;
419 else
420 dev->flags &= ~DEV_OPENED_RW;
421
422 if (need_excl)
423 dev->flags |= DEV_OPENED_EXCL;
424 else
425 dev->flags &= ~DEV_OPENED_EXCL;
426
427 if (!(dev->flags & DEV_REGULAR) &&
428 ((fstat(dev->fd, &buf) < 0) || (buf.st_rdev != dev->dev))) {
429 log_error("%s: fstat failed: Has device name changed?", name);
430 dev_close_immediate(dev);
431 return 0;
432 }
433
434#ifndef O_DIRECT_SUPPORT
435 if (!(dev->flags & DEV_REGULAR))
436 dev_flush(dev);
437#endif
438
439 if ((flags & O_CREAT) && !(flags & O_TRUNC))
440 dev->end = lseek(dev->fd, (off_t) 0, SEEK_END);
441
442 list_add(&_open_devices, &dev->open_list);
443
444 log_debug("Opened %s %s%s%s", dev_name(dev),
445 dev->flags & DEV_OPENED_RW ? "RW" : "RO",
446 dev->flags & DEV_OPENED_EXCL ? " O_EXCL" : "",
447 dev->flags & DEV_O_DIRECT ? " O_DIRECT" : "");
448
449 return 1;
450}

503static int _dev_close(struct device *dev, int immediate)
504{
505 struct lvmcache_info *info;
506
507 if (dev->fd < 0) {
508 log_error("Attempt to close device '%s' "
509 "which is not open.", dev_name(dev));
510 return 0;
511 }
512
513#ifndef O_DIRECT_SUPPORT
514 if (dev->flags & DEV_ACCESSED_W)
515 dev_flush(dev);
516#endif
517
518 if (dev->open_count > 0)
519 dev->open_count--;
520
521 if (immediate && dev->open_count)
522 log_debug("%s: Immediate close attempt while still referenced",
523 dev_name(dev));
524
525 /* Close unless device is known to belong to a locked VG */
526 if (immediate ||
527 (dev->open_count < 1 &&
528 (!(info = info_from_pvid(dev->pvid)) ||
529 !info->vginfo ||
530 !vgname_is_locked(info->vginfo->vgname))))
531 __close(dev);
532
533 return 1;
534}

574/*
575 * Read from 'dev' into 'buf', possibly in 2 distinct regions, denoted
576 * by (offset,len) and (offset2,len2). Thus, the total size of
577 * 'buf' should be len+len2.
578 */
579int dev_read_circular(struct device *dev, uint64_t offset, size_t len,
580 uint64_t offset2, size_t len2, void *buf)
581{
582 if (!dev_read(dev, offset, len, buf)) {
583 log_error("Read from %s failed", dev_name(dev));
584 return 0;
585 }
586
587 /*
588 * The second region is optional, and allows for
589 * a circular buffer on the device.
590 */
591 if (!len2)
592 return 1;
593
594 if (!dev_read(dev, offset2, len2, (char *)buf + len)) {
595 log_error("Circular read from %s failed",
596 dev_name(dev));
597 return 0;
598 }
599
600 return 1;
601}

644int dev_set(struct device *dev, uint64_t offset, size_t len, int value)
645{
646 size_t s;
647 uint32_t pad;
648 char buffer[4096]; // __attribute((aligned(8)));
649
650 if (!dev_open(dev)) {
651 stack;
652 return 0;
653 }
654
655 if ((offset % SECTOR_SIZE) || (len % SECTOR_SIZE))
656 log_debug("Wiping %s at %" PRIu64 " length %" PRIsize_t,
657 dev_name(dev), offset, len);
658 else
659 log_debug("Wiping %s at sector %" PRIu64 " length %" PRIsize_t
660 " sectors", dev_name(dev), offset >> SECTOR_SHIFT,
661 len >> SECTOR_SHIFT);
662
663 memset(buffer, value, sizeof(buffer));
664 while (1) {
665 s = len > sizeof(buffer) ? sizeof(buffer) : len;
666 if (!dev_write(dev, offset, s, buffer))
667 break;
668
669 len -= s;
670 if (!len)
671 break;
672
673 offset += s;
674 }
675
676 dev->flags |= DEV_ACCESSED_W;
677
678 if (!dev_close(dev))
679 stack;
680
681 return (len == 0);
682}
