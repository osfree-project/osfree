208static BOOL acl_init(void)
209{
210 static BOOL initialized, netapi_avail;
211 HMODULE netapi;
212 char buf[256];
213
214 if (initialized)
215 return netapi_avail;
216
217 initialized = TRUE;
218
219 if (DosLoadModule(buf, sizeof(buf), "NETAPI", &netapi))
220 return FALSE;
221
222 if (DosQueryProcAddr(netapi, 0, "NETACCESSGETINFO", (PFN *) &_NetAccessGetInfo) ||
223 DosQueryProcAddr(netapi, 0, "NETACCESSSETINFO", (PFN *) &_NetAccessSetInfo) ||
224 DosQueryProcAddr(netapi, 0, "NETACCESSADD", (PFN *) &_NetAccessAdd))
225 return FALSE;
226
227#if defined(__WATCOMC__) && defined(__386__)
228 NetAccessGetInfo = _NetAccessGetInfo;
229 NetAccessSetInfo = _NetAccessSetInfo;
230 NetAccessAdd = _NetAccessAdd;
231
232// NetAccessGetInfo = (PVOID) (ULONG) (PVOID16) _NetAccessGetInfo;
233// NetAccessSetInfo = (PVOID) (ULONG) (PVOID16) _NetAccessSetInfo;
234// NetAccessAdd = (PVOID) (ULONG) (PVOID16) _NetAccessAdd;
235// (PVOID) (ULONG) (PVOID16)
236#endif
237
238 if ((path = _tmalloc(CCHMAXPATH)) == NULL)
239 return FALSE;
240 if ((data = _tmalloc(ACL_BUFFERSIZE)) == NULL)
241 return FALSE;
242 if ((ai = _tmalloc(sizeof(ACCINFO))) == NULL)
243 return -1;
244
245 netapi_avail = TRUE;
246
247 return netapi_avail;
248}

250static void acl_mkpath(char *buffer, const char *source)
251{
252 char *ptr;
253 static char cwd[CCHMAXPATH];
254 static U_INT cwdlen;
255 U_INT cdrive;
256 ULONG drivemap;
257
258 if (isalpha((int)source[0]) && source[1] == ':')
259 buffer[0] = 0; /* fully qualified names */
260 else
261 {
262 if (cwd[0] == 0)
263 {
264 DosQueryCurrentDisk(&cdrive, &drivemap);
265 cwd[0] = (char)(cdrive + '@');
266 cwd[1] = ':';
267 cwd[2] = '\\';
268 cwdlen = sizeof(cwd) - 3;
269 DosQueryCurrentDir(0, cwd + 3, &cwdlen);
270 cwdlen = strlen(cwd);
271 }
272
273 if (source[0] == '/' || source[0] == '\\')
274 {
275 if (source[1] == '/' || source[1] == '\\')
276 buffer[0] = 0; /* UNC names */
277 else
278 {
279 strncpy(buffer, cwd, 2);
280 buffer[2] = 0;
281 }
282 }
283 else
284 {
285 strcpy(buffer, cwd);
286 if (cwd[cwdlen - 1] != '\\' && cwd[cwdlen - 1] != '/')
287 strcat(buffer, "/");
288 }
289 }
290
291 strcat(buffer, source);
292
293 for (ptr = buffer; *ptr; ptr++)
294 if (*ptr == '/')
295 *ptr = '\\';
296
297 if (ptr[-1] == '\\')
298 ptr[-1] = 0;
299
300 strupr(buffer);
301}

303static int acl_bin2text(char *data, char *text)
304{
305 ACCINFO *ai;
306 ACCLIST *al;
307 U_INT cnt, offs;
308
309 ai = (ACCINFO *) data;
310 al = (ACCLIST *) (data + sizeof(ACCINFO));
311
312 offs = sprintf(text, "ACL1:%X,%d\n",
313 ai -> acc_attr, ai -> acc_count);
314
315 for (cnt = 0; cnt < ai -> acc_count; cnt++)
316 offs += sprintf(text + offs, "%s,%X\n",
317 al[cnt].acl_ugname, al[cnt].acl_access);
318
319 return strlen(text);
320}

357static int acl_text2bin(char *data, char *text, char *path)
358{
359 ACCINFO *ai;
360 ACCLIST *al;
361 char *ptr, *ptr2;
362 U_INT cnt;
363
364 ai = (ACCINFO *) data;
365 ai -> acc_resource_name = PTR16(path);
366
367 if (sscanf(text, "ACL1:%hX,%hd",
368 &ai -> acc_attr, &ai -> acc_count) != 2)
369 return ERROR_INVALID_PARAMETER;
370
371 al = (ACCLIST *) (data + sizeof(ACCINFO));
372 ptr = strchr(text, '\n') + 1;
373
374 for (cnt = 0; cnt < ai -> acc_count; cnt++)
375 {
376 ptr2 = strchr(ptr, ',');
377 strncpy(al[cnt].acl_ugname, ptr, ptr2 - ptr);
378 al[cnt].acl_ugname[ptr2 - ptr] = 0;
379 sscanf(ptr2 + 1, "%hx", &al[cnt].acl_access);
380 ptr = strchr(ptr, '\n') + 1;
381 }
382
383 return sizeof(ACCINFO) + ai -> acc_count * sizeof(ACCLIST);
384}

386int acl_set(char *server, const char *resource, char *buffer)
387{
388 USHORT datalen;
389 PSZ srv = NULL;
390
391 if (!acl_init())
392 return -1;
393
394 if (server)
395 srv = server;
396
397 acl_mkpath(path, resource);
398
399 ai -> acc_resource_name = PTR16(path);
400 ai -> acc_attr = 0;
401 ai -> acc_count = 0;
402
403 NetAccessAdd(srv, 1, ai, sizeof(ACCINFO));
404 /* Ignore any errors, most probably because ACL already exists. */
405 /* In any such case, try updating the existing ACL. */
406
407 datalen = acl_text2bin(data, buffer, path);
408
409 return NetAccessSetInfo(srv, path, 1, data, datalen, 0);
410}
