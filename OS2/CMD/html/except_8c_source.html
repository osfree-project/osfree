2/*
3 *@@sourcefile except.c:
4 * this file contains powerful exception handlers.
5 * except.h also defines easy-to-use macros for them.
6 *
7 * Usage: All OS/2 programs, PM or text mode.
8 *
9 * <B>Introduction</B>
10 *
11 * OS/2 exception handlers are a mess to program and,
12 * if installed wrongly, almost impossible to debug.
13 * The problem is that for any program that does a bit
14 * more than showing a message box, using exception
15 * handlers is a must to avoid system hangs. This
16 * especially applies to multi-thread programs using
17 * mutex semaphores (more on that below). The functions
18 * and macros in here are designed to make that more
19 * simple.
20 *
21 * The macros in except.h automatically insert code for
22 * properly registering and deregistering the handlers
23 * in except.c. You should ALWAYS use these macros
24 * instead of directly registering the handlers to avoid
25 * accidentally forgetting to deregister them. If you
26 * forget to deregister an exception handler, this can
27 * lead to really strange errors (crashes, hangs) which
28 * are nearly impossible to debug because the thread's
29 * stack probably got completely messed up.
30 *
31 * The general idea of these macros is to define
32 * TRY / CATCH blocks similar to C++. If an exception
33 * occurs in the TRY block, execution is transferred to
34 * the CATCH block. (This works in both C and C++, by the
35 * way.)
36 *
37 * The "OnKill" function that was added with V0.9.0 has
38 * been removed again with V0.9.7.
39 *
40 * The general usage is like this:
41 *
42 + int your_protected_func(int ...)
43 + {
44 + TRY_LOUD(excptid) // or: TRY_QUIET(excptid)
45 + {
46 + char *p = NULL;
47 +
48 + .... // the stuff in here is protected by
49 + // the excHandlerLoud or excHandlerQuiet
50 + // exception handler
51 + *p = "A";
52 + }
53 + CATCH(excptid)
54 + {
55 + .... // exception occurred: react here
56 + } END_CATCH(); // always needed!
57 + } // end of your_func
58 *
59 * TRY_LOUD is for installing excHandlerLoud.
60 * TRY_QUIET is for installing excHandlerQuiet.
61 * CATCH / END_CATCH are the same for the two. This
62 * is where the exception handler jumps to if an
63 * exception occurs.
64 * The CATCH block is _required_ even if you do nothing
65 * in there, because the CATCH() macro will deregister
66 * the handler.
67 *
68 * "excptid" can be any C identifier which is not used in
69 * your current variable scope, e.g. "excpt1". This
70 * is used for creating an EXCEPTSTRUCT variable of
71 * that name on the stack. The "excptid"'s in TRY_* and
72 * CATCH must match, since this is where the macros
73 * store the exception handler data.
74 *
75 * These macros may be nested if you use different
76 * "excptid"'s for sub-macros.
77 *
78 * Inside the TRY and CATCH blocks, you must not use
79 * "goto" (to a location outside the block) or "return",
80 * because this will not deregister the handler.
81 *
82 * Keep in mind that all the code in the TRY_* block is
83 * protected by the handler, including all functions that
84 * get called. So if you enclose your main() code in a
85 * TRY_* block, your entire application is protected.
86 * If any subfunction fails, execution is transferred to
87 * the closest CATCH() that was installed (as with C++
88 * try and catch).
89 *
90 * <B>Asynchronous exceptions</B>
91 *
92 * The exception handlers in this file (which are installed
93 * with the TRY/CATCH mechanism) only intercept synchronous
94 * exceptions, most importantly, XCPT_ACCESS_VIOLATION (see
95 * excHandlerLoud for a list). They do not protect your code
96 * against asynchronous exceptions.
97 *
98 * OS/2 defines asynchronous exceptions to be those that
99 * can be delayed. With OS/2, there are only three of these:
100 *
101 * -- XCPT_PROCESS_TERMINATE
102 * -- XCPT_ASYNC_PROCESS_TERMINATE
103 * -- XCPT_SIGNAL (thread 1 only)
104 *
105 * To protect yourself against these also, put the section
106 * in question in a DosEnterMustComplete/DosExitMustComplete
107 * block as well.
108 *
109 * <B>Mutex semaphores</B>
110 *
111 * The problem with OS/2 mutex semaphores is that they are
112 * sometimes not automatically released when a thread terminates.
113 * If there are several mutexes involved and they are released
114 * in improper order, you can get zombie threads on exit.
115 * Even worse, if this happens to a PM thread, this will hang
116 * the system.
117 *
118 * As a result, you should protect any section of code which
119 * requests a semaphore with the exception handlers.
120 *
121 * So _whenever_ you request a mutex semaphore, enclose
122 * the block with TRY/CATCH in case the code crashes.
123 * Besides, enclose the TRY/CATCH block in a must-complete
124 * section, like this:
125 *
126 + HMTX hmtx = ...
127 +
128 + int your_func(int)
129 + {
130 + volatile BOOL fSemOwned = FALSE;
131 +
132 + TRY_QUIET(excpt1) // or TRY_LOUD
133 + {
134 + if (fSemOwned = !DosRequestMutexSem(hmtx, ...))
135 + { ... // work on your protected data
136 + }
137 + // mutex gets released below
138 + }
139 + CATCH(excpt1) { } END_CATCH(); // always needed!
140 +
141 + if (fSemOwned)
142 + // this gets executed always, even if an exception occurred
143 + DosReleaseMutexSem(hmtx);
144 + } // end of your_func
145 *
146 * This way your mutex semaphore gets released in every
147 * possible condition.
148 *
149 * <B>Customizing</B>
150 *
151 * As opposed to versions before 0.9.0, this code is now
152 * completely independent of XWorkplace. This file now
153 * contains "pure" exception handlers only.
154 *
155 * However, you can customize these exception handlers by
156 * calling excRegisterHooks. This is what XWorkplace does now.
157 * This should be done upon initialization of your application.
158 * If excRegisterHooks is not called, the following safe
159 * defaults are used:
160 *
161 * -- the trap log file is TRAP.LOG in the root
162 * directory of your boot drive.
163 *
164 * For details on the provided exception handlers, refer
165 * to excHandlerLoud and excHandlerQuiet.
166 *
167 * More useful debug information can be found in the "OS/2 Debugging
168 * Handbook", which is now available in INF format on the IBM
169 * DevCon site ("http://service2.boulder.ibm.com/devcon/").
170 * This book shows worked examples of how to unwind a stack dump.
171 *
172 * This file incorporates code from the following:
173 * -- Monte Copeland, IBM Boca Ration, Florida, USA (1993)
174 * -- Roman Stangl, from the Program Commander/2 sources
175 * (1997-98)
176 * -- Marc Fiammante, John Currier, Kim Rasmussen,
177 * Anthony Cruise (EXCEPT3.ZIP package for a generic
178 * exception handling DLL, available at Hobbes).
179 *
180 * If not explicitly stated otherwise, the code has been written
181 * by me, Ulrich M”ller.
182 *
183 * Note: Version numbering in this file relates to XWorkplace version
184 * numbering.
185 *
186 *@@header "helpers\except.h"
187 */

275VOID excDescribePage(FILE *file, ULONG ulCheck)
276{
277 APIRET arc;
278 ULONG ulCountPages = 1;
279 ULONG ulFlagsPage = 0;
280 arc = DosQueryMem((PVOID)ulCheck, &ulCountPages, &ulFlagsPage);
281
282 if (arc == NO_ERROR)
283 {
284 fprintf(file, "valid, flags: ");
285 if (ulFlagsPage & PAG_READ)
286 fprintf(file, "read ");
287 if (ulFlagsPage & PAG_WRITE)
288 fprintf(file, "write ");
289 if (ulFlagsPage & PAG_EXECUTE)
290 fprintf(file, "execute ");
291 if (ulFlagsPage & PAG_GUARD)
292 fprintf(file, "guard ");
293 if (ulFlagsPage & PAG_COMMIT)
294 fprintf(file, "committed ");
295 if (ulFlagsPage & PAG_SHARED)
296 fprintf(file, "shared ");
297 if (ulFlagsPage & PAG_FREE)
298 fprintf(file, "free ");
299 if (ulFlagsPage & PAG_BASE)
300 fprintf(file, "base ");
301 }
302 else if (arc == ERROR_INVALID_ADDRESS)
303 fprintf(file, "invalid");
304}

315VOID excPrintStackFrame(FILE *file, // in: output log file
316 PSZ pszDescription, // in: description for stack frame (should be eight chars)
317 ULONG ulAddress) // in: address to debug
318{
319 APIRET arc = NO_ERROR;
320 HMODULE hmod1 = NULLHANDLE;
321 CHAR szMod1[2*CCHMAXPATH] = "unknown";
322 ULONG ulObject = 0,
323 ulOffset = 0;
324 fprintf(file,
325 " %-8s: %08lX ",
326 pszDescription,
327 ulAddress);
328 arc = DosQueryModFromEIP(&hmod1,
329 &ulObject,
330 sizeof(szMod1), szMod1,
331 &ulOffset,
332 ulAddress);
333
334 if (arc != NO_ERROR)
335 {
336 // error:
337 fprintf(file,
338 " %-8s Error: DosQueryModFromEIP returned %lu\n",
339 szMod1,
340 arc);
341 }
342 else
343 {
344 CHAR szFullName[2*CCHMAXPATH] = "unknown";
345
346 fprintf(file,
347 " %-8s %02lX:%08lX\n ",
348 szMod1,
349 ulObject + 1, // V0.9.12 (2001-05-12) [umoeller]
350 ulOffset); // V0.9.12 (2001-05-12) [umoeller]
351
352 DosQueryModuleName(hmod1, sizeof(szFullName), szFullName);
353 dbgPrintStackFrame(file,
354 szFullName,
355 ulObject,
356 ulOffset);
357
358 fprintf(file, "\n");
359
360 // make a 'tick' sound to let the user know we're still alive
361 DosBeep(2000, 10);
362 }
363}

365/*
366 *@@ excDumpStackFrames:
367 * called from excExplainException to dump the
368 * thread's stack frames. This calls excPrintStackFrame
369 * for each stack frame found.
370 *
371 *@@added V0.9.4 (2000-06-15) [umoeller]
372 */
373
374VOID excDumpStackFrames(FILE *file, // in: logfile from fopen()
375 PTIB ptib,
376 PCONTEXTRECORD pContextRec) // in: excpt info
377{
378 PULONG pulStackWord = 0;
379
380 fprintf(file, "\n\nStack frames:\n Address Module seg:ofs\n");
381
382 // first the trapping address itself
383 excPrintStackFrame(file,
384 "CS:EIP ",
385 pContextRec->ctx_RegEip);
386
387
388 pulStackWord = (PULONG)pContextRec->ctx_RegEbp;
389 /* if (pContextRec->ctx_RegEbp < pContextRec->ctx_RegEsp)
390 pulStackWord = (PULONG)(pContextRec->ctx_RegEbp & 0xFFFFFFF0);
391 else
392 pulStackWord = (PULONG)(pContextRec->ctx_RegEsp & 0xFFFFFFF0); */
393
394 while ( (pulStackWord != 0)
395 && (pulStackWord < (PULONG)ptib->tib_pstacklimit)
396 )
397 {
398 CHAR szAddress[20];
399
400 if (((ULONG)pulStackWord & 0x00000FFF) == 0x00000000)
401 {
402 // we're on a page boundary: check access
403 ULONG ulCountPages = 0x1000;
404 ULONG ulFlagsPage = 0;
405 APIRET arc = DosQueryMem((void *)pulStackWord,
406 &ulCountPages,
407 &ulFlagsPage);
408 if ( (arc != NO_ERROR)
409 || ( (arc == NO_ERROR)
410 && ( !( ((ulFlagsPage & (PAG_COMMIT|PAG_READ))
411 == (PAG_COMMIT|PAG_READ)
412 )
413 )
414 )
415 )
416 )
417 {
418 fprintf(file, "\n %08lX: ", (ULONG)pulStackWord);
419 fprintf(file, "Page inaccessible");
420 pulStackWord += 0x1000;
421 continue; // for
422 }
423 }
424
425 sprintf(szAddress, "%08lX",
426 (ULONG)pulStackWord);
427 excPrintStackFrame(file,
428 szAddress,
429 *(pulStackWord+1));
430 pulStackWord = (PULONG)*(pulStackWord);
431
432 if (pulStackWord == 0)
433 fprintf(file, "\n pulStackWord == 0");
434 else if (pulStackWord >= (PULONG)ptib->tib_pstacklimit)
435 fprintf(file, "\n pulStackWord >= (PULONG)ptib->tib_pstacklimit");
436 } // end while
437}

458VOID excExplainException(FILE *file, // in: logfile from fopen()
459 PSZ pszHandlerName, // in: descriptive string
460 PEXCEPTIONREPORTRECORD pReportRec, // in: excpt info
461 PCONTEXTRECORD pContextRec) // in: excpt info
462{
463 ULONG aulBuf[3];
464 const char *pcszVersion = "unknown";
465
466 PTIB ptib = NULL;
467 PPIB ppib = NULL;
468 HMODULE hMod1 = NULLHANDLE, hMod2 = NULLHANDLE;
469 CHAR szMod1[CCHMAXPATH] = "unknown",
470 szMod2[CCHMAXPATH] = "unknown";
471 ULONG ulObjNum = -1,
472 ulOffset = -1;
473 ULONG ul;
474
475 ULONG ulOldPriority = 0x0100; // regular, delta 0
476
477 // raise global flag for whether this func is running
478 // V0.9.13 (2001-06-19) [umoeller]
479 G_ulExplainExceptionRunning++;
480
481 // raise this thread's priority, because this
482 // might take some time
483 if (DosGetInfoBlocks(&ptib, &ppib) == NO_ERROR)
484 if (ptib)
485 if (ptib->tib_ptib2)
486 {
487 ulOldPriority = ptib->tib_ptib2->tib2_ulpri;
488 DosSetPriority(PRTYS_THREAD,
489 PRTYC_REGULAR,
490 PRTYD_MAXIMUM,
491 0); // current thread
492 }
493
494 // make some noise
495#ifndef __NOEXCEPTIONBEEPS__ // V0.9.19 (2002-04-17) [umoeller]
496 if (G_fBeepOnException)
497 {
498 DosBeep( 250, 30);
499 DosBeep( 500, 30);
500 DosBeep(1000, 30);
501 DosBeep(2000, 30);
502 DosBeep(4000, 30);
503 DosBeep(2000, 30);
504 DosBeep(1000, 30);
505 DosBeep( 500, 30);
506 DosBeep( 250, 30);
507 }
508#endif
509
510 // generic exception info
511 DosQuerySysInfo(QSV_VERSION_MAJOR, // 11
512 QSV_VERSION_REVISION, // 13 V1.0.0 (2002-08-28) [umoeller]
513 &aulBuf, sizeof(aulBuf));
514 // Warp 3 is reported as 20.30
515 // Warp 4 is reported as 20.40
516 // Aurora is reported as 20.45
517
518 if (aulBuf[0] == 20)
519 {
520 switch (aulBuf[1])
521 {
522 case 30: pcszVersion = "Warp 3"; break;
523 case 40: pcszVersion = "Warp 4"; break;
524 case 45: pcszVersion = "WSeB kernel"; break;
525 }
526 }
527 fprintf(file,
528 "Running OS/2 version: %u.%u.%u (%s)\n",
529 aulBuf[0], // major
530 aulBuf[1],
531 aulBuf[2], // revision V1.0.0 (2002-08-28) [umoeller]
532 pcszVersion);
533
534
535 // generic exception info
536 fprintf(file,
537 "\n%s:\n Exception type: %08lX\n Address: %08lX\n Params: ",
538 pszHandlerName,
539 pReportRec->ExceptionNum,
540 (ULONG)pReportRec->ExceptionAddress);
541 for (ul = 0; ul < pReportRec->cParameters; ul++)
542 {
543 fprintf(file, "%08lX ",
544 pReportRec->ExceptionInfo[ul]);
545 }
546
547 // now explain the exception in a bit more detail;
548 // depending on the exception, pReportRec->ExceptionInfo
549 // contains some useful data
550 switch (pReportRec->ExceptionNum)
551 {
552 case XCPT_ACCESS_VIOLATION:
553 fprintf(file, "\nXCPT_ACCESS_VIOLATION: ");
554 if (pReportRec->ExceptionInfo[0] & XCPT_READ_ACCESS)
555 fprintf(file, "XCPT_READ_ACCESS -- Invalid read access from 0x%04lX:%08lX.\n",
556 pContextRec->ctx_SegDs,
557 pReportRec->ExceptionInfo[1]);
558 else if (pReportRec->ExceptionInfo[0] & XCPT_WRITE_ACCESS)
559 fprintf(file, "XCPT_WRITE_ACCESS -- Invalid write access to 0x%04lX:%08lX.\n",
560 pContextRec->ctx_SegDs,
561 pReportRec->ExceptionInfo[1]);
562 else if (pReportRec->ExceptionInfo[0] & XCPT_SPACE_ACCESS)
563 fprintf(file, "XCPT_SPACE_ACCESS -- Invalid space access at 0x%04lX.\n",
564 pReportRec->ExceptionInfo[1]);
565 else if (pReportRec->ExceptionInfo[0] & XCPT_LIMIT_ACCESS)
566 fprintf(file, "XCPT_LIMIT_ACCESS -- Invalid limit access occurred.\n");
567 else if (pReportRec->ExceptionInfo[0] == XCPT_UNKNOWN_ACCESS)
568 fprintf(file, "XCPT_UNKNOWN_ACCESS -- unknown at 0x%04lX:%08lX\n",
569 pContextRec->ctx_SegDs,
570 pReportRec->ExceptionInfo[1]);
571 fprintf(file,
572 "Explanation: An attempt was made to access a memory object which does\n"
573 " not belong to the current process. Most probable causes\n"
574 " for this are that an invalid pointer was used, there was\n"
575 " confusion with administering memory or error conditions \n"
576 " were not properly checked for.\n");
577 break;
578
579 case XCPT_INTEGER_DIVIDE_BY_ZERO:
580 fprintf(file, "\nXCPT_INTEGER_DIVIDE_BY_ZERO.\n");
581 fprintf(file,
582 "Explanation: An attempt was made to divide an integer value by zero,\n"
583 " which is not defined.\n");
584 break;
585
586 case XCPT_ILLEGAL_INSTRUCTION:
587 fprintf(file, "\nXCPT_ILLEGAL_INSTRUCTION.\n");
588 fprintf(file,
589 "Explanation: An attempt was made to execute an instruction that\n"
590 " is not defined on this machine's architecture.\n");
591 break;
592
593 case XCPT_PRIVILEGED_INSTRUCTION:
594 fprintf(file, "\nXCPT_PRIVILEGED_INSTRUCTION.\n");
595 fprintf(file,
596 "Explanation: An attempt was made to execute an instruction that\n"
597 " is not permitted in the current machine mode or that\n"
598 " the program had no permission to execute.\n");
599 break;
600
601 case XCPT_INTEGER_OVERFLOW:
602 fprintf(file, "\nXCPT_INTEGER_OVERFLOW.\n");
603 fprintf(file,
604 "Explanation: An integer operation generated a carry-out of the most\n"
605 " significant bit. This is a sign of an attempt to store\n"
606 " a value which does not fit into an integer variable.\n");
607 break;
608
609 default:
610 fprintf(file, "\nUnknown OS/2 exception number %d.\n", pReportRec->ExceptionNum);
611 fprintf(file, "Look this up in the OS/2 header files.\n");
612 break;
613 }
614
615 // V0.9.16 (2001-11-02) [pr]: We already got this info. above - this overwrites the
616 // original values before the priority change, which is rather confusing.
617 // if (DosGetInfoBlocks(&ptib, &ppib) == NO_ERROR)
618 {
619 /*
620 * process info:
621 *
622 */
623
624 if ((ptib) && (ppib)) // (99-11-01) [umoeller]
625 {
626 if (pContextRec->ContextFlags & CONTEXT_CONTROL)
627 {
628 // get the main module
629 hMod1 = ppib->pib_hmte;
630 DosQueryModuleName(hMod1,
631 sizeof(szMod1),
632 szMod1);
633
634 // get the trapping module
635 DosQueryModFromEIP(&hMod2,
636 &ulObjNum,
637 sizeof(szMod2),
638 szMod2,
639 &ulOffset,
640 pContextRec->ctx_RegEip);
641 DosQueryModuleName(hMod2,
642 sizeof(szMod2),
643 szMod2);
644 }
645
646 fprintf(file,
647 "\nProcess information:"
648 "\n Process ID: 0x%lX"
649 "\n Process module: 0x%lX (%s)"
650 "\n Trapping module: 0x%lX (%s)"
651 "\n Object: %ld\n", // V0.9.16 (2001-11-02) [pr]: make this display signed
652 ppib->pib_ulpid,
653 hMod1, szMod1,
654 hMod2, szMod2,
655 ulObjNum);
656
657 fprintf(file,
658 "\nTrapping thread information:"
659 "\n Thread ID: 0x%lX (%lu)"
660 "\n Thread slot ID: 0x%lX (%lu)" // added V0.9.19 (2002-03-28) [umoeller]
661 "\n Priority: 0x%lX\n",
662 ptib->tib_ptib2->tib2_ultid, ptib->tib_ptib2->tib2_ultid,
663 ptib->tib_ordinal, ptib->tib_ordinal,
664 ulOldPriority);
665 }
666 else
667 fprintf(file, "\nProcess information was not available.");
668
669 /*
670 * now call the hook, if one has been defined,
671 * so that the application can write additional
672 * information to the traplog (V0.9.0)
673 */
674
675 if (G_pfnExcHook)
676 G_pfnExcHook(file, ptib, ulOldPriority); // V0.9.16 (2001-12-02) [pr]
677
678 // *** registers
679
680 fprintf(file, "\nRegisters:");
681 if (pContextRec->ContextFlags & CONTEXT_INTEGER)
682 {
683 // DS the following 4 added V0.9.6 (2000-11-06) [umoeller]
684 fprintf(file, "\n DS = %08lX ", pContextRec->ctx_SegDs);

777/*
778 *@@ excRegisterHooks:
779 * this registers hooks which get called for
780 * exception handlers. You can set any of the
781 * hooks to NULL for safe defaults (see top of
782 * except.c for details). You can set none,
783 * one, or both of the hooks, and you can call
784 * this function several times.
785 *
786 * Both hooks get called whenever an exception
787 * occurs, so there better be no bugs in these
788 * routines. ;-) They only get called from
789 * within excHandlerLoud (because excHandlerQuiet
790 * writes no trap logs).
791 *
792 * Registering hooks affects all threads that use
793 * the exception handlers.
794 *
795 * The hooks are as follows:
796 *
797 * -- pfnExcOpenFileNew gets called to open
798 * the trap log file. This can return a FILE*
799 * pointer from fopen() (which will be closed
800 * automatically by the handler).
801 *
802 * If this hook is not defined, ?:\TRAP.LOG is
803 * used. Use this hook to specify a different file
804 * and have some notes written into it before the
805 * actual exception info.
806 *
807 * If the hook returns a null FILE* pointer,
808 * logging is disabled, and the "loud" handler
809 * effectively behaves like the "quiet" handler.
810 *
811 * -- pfnExcHookNew gets called while the trap log
812 * is being written. At this point,
813 * the following info has been written into
814 * the trap log already:
815 *
816 * -- exception type/address block
817 *
818 * -- exception explanation
819 *
820 * -- process information
821 *
822 * _After_ the hook, the exception handler
823 * continues with the "Registers" information
824 * and stack dump/analysis.
825 *
826 * Use this hook to write additional application
827 * info into the trap log, such as the state
828 * of your own threads and mutexes.
829 *
830 * -- pfnExcHookError gets called when the TRY_* macros
831 * fail to install an exception handler (when
832 * DosSetExceptionHandler fails). I've never seen
833 * this happen.
834 *
835 *@@added V0.9.0 [umoeller]
836 *@@changed V0.9.2 (2000-03-10) [umoeller]: pfnExcHookError added
837 */
838

840 PFNEXCHOOK pfnExcHookNew,

851/*
852 *@@ excHandlerLoud:
853 * this is the "sophisticated" exception handler;
854 * which gives forth a loud sequence of beeps thru the
855 * speaker, writes a trap log and then returns back
856 * to the thread to continue execution, i.e. the
857 * default OS/2 exception handler will never get
858 * called.
859 *
860 * This requires a setjmp() call on
861 * EXCEPTIONREGISTRATIONRECORD2.jmpThread before
862 * being installed. The TRY_LOUD macro will take
863 * care of this for you (see except.c).
864 *
865 * This intercepts the following exceptions (see
866 * the OS/2 Control Program Reference for details):
867 *
868 * -- XCPT_ACCESS_VIOLATION (traps 0x0d, 0x0e)
869 * -- XCPT_INTEGER_DIVIDE_BY_ZERO (trap 0)
870 * -- XCPT_ILLEGAL_INSTRUCTION (trap 6)
871 * -- XCPT_PRIVILEGED_INSTRUCTION
872 * -- XCPT_INTEGER_OVERFLOW (trap 4)
873 *
874 * For these exceptions, we call the functions in debug.c
875 * to try to find debug code or SYM file information about
876 * what source code corresponds to the error.
877 *
878 * See excRegisterHooks for the default setup of this.
879 *
880 * Note that to get meaningful debugging information
881 * in this handler's traplog, you need the following:
882 *
883 * a) have a MAP file created at link time (/MAP)
884 *
885 * b) convert the MAP to a SYM file using MAPSYM
886 *
887 * c) put the SYM file in the same directory of
888 * the module (EXE or DLL). This must have the
889 * same filestem as the module.
890 *
891 * All other exceptions are passed to the next handler
892 * in the exception handler chain. This might be the
893 * C/C++ compiler handler or the default OS/2 handler,
894 * which will probably terminate the process.
895 *
896 *@@changed V0.9.0 [umoeller]: added support for thread termination
897 *@@changed V0.9.2 (2000-03-10) [umoeller]: switched date format to ISO
898 *@@changed V0.9.19 (2002-05-07) [umoeller]: added EXCEPTIONREPORTRECORD info so that catch block can check that
899 */
900
901ULONG _System excHandlerLoud(PEXCEPTIONREPORTRECORD pReportRec,
902 PEXCEPTIONREGISTRATIONRECORD2 pRegRec2,
903 PCONTEXTRECORD pContextRec,
904 PVOID pv)
905{
906 /* From the VAC++3 docs:
907 * "The first thing an exception handler should do is check the
908 * exception flags. If EH_EXIT_UNWIND is set, meaning
909 * the thread is ending, the handler tells the operating system
910 * to pass the exception to the next exception handler. It does the
911 * same if the EH_UNWINDING flag is set, the flag that indicates
912 * this exception handler is being removed.
913 * The EH_NESTED_CALL flag indicates whether the exception
914 * occurred within an exception handler. If the handler does
915 * not check this flag, recursive exceptions could occur until
916 * there is no stack remaining."
917 *
918 * So for all these conditions, we exit immediately.
919 */
920
921 // XWP V1.0.4 (2005-10-09) [pr]: Optimize
922 if (pReportRec->fHandlerFlags & (EH_EXIT_UNWIND | EH_UNWINDING | EH_NESTED_CALL))
923 return XCPT_CONTINUE_SEARCH;
924
925 switch (pReportRec->ExceptionNum)
926 {
927 case XCPT_ACCESS_VIOLATION:
928 case XCPT_INTEGER_DIVIDE_BY_ZERO:
929 case XCPT_ILLEGAL_INSTRUCTION:
930 case XCPT_PRIVILEGED_INSTRUCTION:
931 case XCPT_INVALID_LOCK_SEQUENCE:
932 case XCPT_INTEGER_OVERFLOW:
933 {
934 // "real" exceptions:
935 FILE *file = NULL;
936
937 // open traplog file;
938 if (G_pfnExcOpenFile)
939 // hook defined for this: call it
940 file = G_pfnExcOpenFile();
941 else
942 {
943 CHAR szFileName[100];
944 // no hook defined: open some
945 // default traplog file in root directory of
946 // boot drive
947 sprintf(szFileName, "%c:\\trap.log", doshQueryBootDrive());
948
949 if (file = fopen(szFileName, "a"))
950 {
951 DATETIME DT;
952 DosGetDateTime(&DT);
953 fprintf(file,
954 "\nTrap message -- Date: %04d-%02d-%02d, Time: %02d:%02d:%02d\n",
955 DT.year, DT.month, DT.day,
956 DT.hours, DT.minutes, DT.seconds);
957 fprintf(file, "------------------------------------------------\n");
958
959 }
960 }
961
962 if (file)
963 {
964 // write error log
965 excExplainException(file,
966 "excHandlerLoud",
967 pReportRec,
968 pContextRec);
969 fclose(file);
970 }
971
972 // copy report rec to user buffer
973 // V0.9.19 (2002-05-07) [umoeller]
974 memcpy(&pRegRec2->err,
975 pReportRec,
976 sizeof(EXCEPTIONREPORTRECORD));
977
978 // jump back to failing routine
979 longjmp(pRegRec2->jmpThread, pReportRec->ExceptionNum);
980 }
981 break;
982 }
983
984 // not handled
985 return XCPT_CONTINUE_SEARCH;
986}
