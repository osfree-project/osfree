185static int _attrib( ATTRIB_STRUCT *Attrib )
186{
187 unsigned int old_attrib;
188 int fval, rval = 0, rc;
189 char *source_name;
190 FILESEARCH dir;
191
192 EnableSignals();
193
194 // copy date/time range info
195 memmove( &(dir.aRanges), &aRanges, sizeof(RANGES) );
196
197 // change attributes
198 for ( fval = FIND_FIRST; ( find_file( fval, Attrib->szSource, ( Attrib->nMode | FIND_BYATTS | FIND_DATERANGE | FIND_NO_DOTNAMES ), &dir, Attrib->szTarget ) != NULL ); fval = FIND_NEXT) {
199
200 if (( setjmp( cv.env ) == -1 ) || ( cv.exception_flag )) {
201 (void)DosFindClose( dir.hdir );
202 return CTRLC;
203 }
204
205 // display old attributes, new attributes, and filename
206 old_attrib = ( dir.attrib & 0x37 );
207 dir.attrib = (unsigned char)(( old_attrib & Attrib->cAnd ) | Attrib->cOr );
208
209 // if attributes were changed, show new ones
210 if (( Attrib->fFlags & ATTRIB_QUIET ) == 0 ) {
211
212 qputs( show_atts( old_attrib ));
213
214 if (( Attrib->cAnd != 0x37 ) || ( Attrib->cOr != 0 ))
215 printf( " -> %s", show_atts( dir.attrib ));
216
217 printf(" %s", Attrib->szTarget );
218 crlf();
219 _page_break();
220 }
221
222 // check for a change; ignore if same
223 // (can't change directory attribute)
224 if (( old_attrib != (unsigned int)dir.attrib ) && (( rc = SetFileMode( Attrib->szTarget, (dir.attrib & 0x27))) != 0 ))
225 rval = error( rc, Attrib->szTarget );
226 }
227
228 // modify matching subdirectory files too?
229 if ( Attrib->fFlags & ATTRIB_SUBDIRS ) {
230
231 // strip the filename & add wildcards for directory search
232 sprintf( Attrib->szSource, FMT_DOUBLE_STR, path_part( Attrib->szSource ), WILD_FILE );
233
234 // save the current subdirectory start
235 source_name = strchr( Attrib->szSource, '*' );
236
237 // search for all subdirectories in this (sub)directory
238 // tell find_file() not to display errors, & to only
239 // return subdir names
240 for ( fval = FIND_FIRST; ( find_file( fval, Attrib->szSource, ( Attrib->nMode | 0x310 ),&dir,NULL ) != NULL ); fval = FIND_NEXT) {
241
242 // make the new "source"

246 rval = _attrib( Attrib );
247 if (( setjmp( cv.env ) == -1 ) || ( rval == CTRLC )) {
248 // reset the directory search handle
249 (void)DosFindClose( dir.hdir );
250 return CTRLC;
251 }
252
253 // restore the original name
254 strcpy( source_name, WILD_FILE );
255 }
256 }
257

259 HoldSignals();
260
261 return rval;
262}
