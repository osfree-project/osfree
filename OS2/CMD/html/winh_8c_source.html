425/*
426 *@@ winhInsertMenuItem:
427 * this inserts one one menu item into a given menu.
428 *
429 * Returns the return value of the MM_INSERTITEM msg:
430 * -- MIT_MEMERROR: space allocation for menu item failed
431 * -- MIT_ERROR: other error
432 * -- other: zero-based index of new item in menu.
433 */
434
435SHORT winhInsertMenuItem(HWND hwndMenu, // in: menu to insert item into
436 SHORT iPosition, // in: zero-based index of where to
437 // insert or MIT_END
438 SHORT sItemId, // in: ID of new menu item
439 const char *pcszItemTitle, // in: title of new menu item
440 SHORT afStyle,
441 // in: MIS_* style flags.
442 // Valid menu item styles are:
443 // -- MIS_SUBMENU
444 // -- MIS_SEPARATOR
445 // -- MIS_BITMAP: the display object is a bit map.
446 // -- MIS_TEXT: the display object is a text string.
447 // -- MIS_BUTTONSEPARATOR:
448 // The item is a menu button. Any menu can have zero,
449 // one, or two items of this type. These are the last
450 // items in a menu and are automatically displayed after
451 // a separator bar. The user cannot move the cursor to
452 // these items, but can select them with the pointing
453 // device or with the appropriate key.
454 // -- MIS_BREAK: the item begins a new row or column.
455 // -- MIS_BREAKSEPARATOR:
456 // Same as MIS_BREAK, except that it draws a separator
457 // between rows or columns of a pull-down menu.
458 // This style can only be used within a submenu.
459 // -- MIS_SYSCOMMAND:
460 // menu posts a WM_SYSCOMMAND message rather than a
461 // WM_COMMAND message.
462 // -- MIS_OWNERDRAW:
463 // WM_DRAWITEM and WM_MEASUREITEM notification messages
464 // are sent to the owner to draw the item or determine its size.
465 // -- MIS_HELP:
466 // menu posts a WM_HELP message rather than a
467 // WM_COMMAND message.
468 // -- MIS_STATIC
469 // This type of item exists for information purposes only.
470 // It cannot be selected with the pointing device or
471 // keyboard.
472 SHORT afAttr)
473 // in: MIA_* attribute flags
474 // Valid menu item attributes (afAttr) are:
475 // -- MIA_HILITED: if and only if, the item is selected.
476 // -- MIA_CHECKED: a check mark appears next to the item (submenu only).
477 // -- MIA_DISABLED: item is disabled and cannot be selected.
478 // The item is drawn in a disabled state (gray).
479 // -- MIA_FRAMED: a frame is drawn around the item (top-level menu only).
480 // -- MIA_NODISMISS:
481 // if the item is selected, the submenu remains down. A menu
482 // with this attribute is not hidden until the application
483 // or user explicitly does so, for example by selecting either
484 // another menu on the action bar or by pressing the escape key.
485{
486 MENUITEM mi;
487
488 mi.iPosition = iPosition;
489 mi.afStyle = afStyle;
490 mi.afAttribute = afAttr;
491 mi.id = sItemId;
492 mi.hwndSubMenu = 0;
493 mi.hItem = 0;
494
495 return SHORT1FROMMR(WinSendMsg(hwndMenu,
496 MM_INSERTITEM,
497 (MPARAM)&mi,
498 (MPARAM)pcszItemTitle));
499}

501/*
502 *@@ winhInsertSubmenu:
503 * this inserts a submenu into a given menu and, if
504 * sItemId != 0, inserts one item into this new submenu also.
505 *
506 * See winhInsertMenuItem for valid menu item styles and
507 * attributes.
508 *
509 * Returns the HWND of the new submenu.
510 */
511
512HWND winhInsertSubmenu(HWND hwndMenu, // in: menu to add submenu to
513 ULONG iPosition, // in: index where to add submenu or MIT_END
514 SHORT sMenuId, // in: menu ID of new submenu
515 const char *pcszSubmenuTitle, // in: title of new submenu
516 USHORT afMenuStyle, // in: MIS* style flags for submenu;
517 // MIS_SUBMENU will always be added
518 SHORT sItemId, // in: ID of first item to add to submenu;
519 // if 0, no first item is inserted
520 const char *pcszItemTitle, // in: title of this item
521 // (if sItemID != 0)
522 USHORT afItemStyle, // in: style flags for this item, e.g. MIS_TEXT
523 // (this is ignored if sItemID == 0)
524 USHORT afAttribute) // in: attributes for this item, e.g. MIA_DISABLED
525 // (this is ignored if sItemID == 0)
526{
527 MENUITEM mi;
528 SHORT src = MIT_ERROR;
529 HWND hwndNewMenu;
530
531 // create new, empty menu
532 hwndNewMenu = WinCreateMenu(hwndMenu,
533 NULL); // no menu template
534 if (hwndNewMenu)
535 {
536 // add "submenu item" to this empty menu;
537 // for some reason, PM always needs submenus
538 // to be a menu item
539 mi.iPosition = iPosition;
540 mi.afStyle = afMenuStyle | MIS_SUBMENU;
541 mi.afAttribute = 0;
542 mi.id = sMenuId;
543 mi.hwndSubMenu = hwndNewMenu;
544 mi.hItem = 0;
545 src = SHORT1FROMMR(WinSendMsg(hwndMenu, MM_INSERTITEM, (MPARAM)&mi, (MPARAM)pcszSubmenuTitle));
546 if ( (src != MIT_MEMERROR)
547 && (src != MIT_ERROR)
548 )
549 {
550 // set the new menu's ID to the same as the
551 // submenu item
552 WinSetWindowUShort(hwndNewMenu, QWS_ID, sMenuId);
553
554 if (sItemId)
555 {
556 // item id given: insert first menu item also
557 mi.iPosition = 0;
558 mi.afStyle = afItemStyle;
559 mi.afAttribute = afAttribute;
560 mi.id = sItemId;
561 mi.hwndSubMenu = 0;
562 mi.hItem = 0;
563 WinSendMsg(hwndNewMenu,
564 MM_INSERTITEM,
565 (MPARAM)&mi,
566 (MPARAM)pcszItemTitle);
567 }
568 }
569 }
570 return hwndNewMenu;
571}

573/*
574 *@@ winhSetMenuCondCascade:
575 * sets the "conditional cascade" style
576 * on the specified submenu.
577 *
578 * This style must always be enabled manually
579 * because the resource compiler won't handle it.
580 *
581 * Note: Pass in the _submenu_ window handle,
582 * not the one of the parent. With lDefaultItem,
583 * specify the item ID in the submenu which is
584 * to be checked as the default item.
585 *
586 *@@added V0.9.12 (2001-05-22) [umoeller]
587 *@@changed V0.9.20 (2002-08-10) [umoeller]: now supporting calling this more than once
588 */
589
590BOOL winhSetMenuCondCascade(HWND hwndMenu, // in: submenu handle
591 LONG lDefaultItem) // in: item ID of new default item
592{
593 BOOL brc;
594 ULONG ulStyle = WinQueryWindowULong(hwndMenu, QWL_STYLE);
595 LONG lOldDefault = -1;
596
597 if (ulStyle & MS_CONDITIONALCASCADE)
598 {
599 // menu is already conditional cascade:
600 lOldDefault = (LONG)WinSendMsg(hwndMenu,
601 MM_QUERYDEFAULTITEMID,
602 0,
603 0);
604 }
605 else
606 {
607 ulStyle |= MS_CONDITIONALCASCADE;
608 WinSetWindowULong(hwndMenu, QWL_STYLE, ulStyle);
609 }
610
611 // make the first item in the subfolder
612 // the default of cascading submenu
613 brc = (BOOL)WinSendMsg(hwndMenu,
614 MM_SETDEFAULTITEMID,
615 (MPARAM)lDefaultItem,
616 0);
617
618 if ( (lOldDefault != -1)
619 && (lOldDefault != lDefaultItem)
620 )
621 {
622 // unset the "checked" attribute of the old one
623 // or we'll have two in the menu
624 WinSendMsg(hwndMenu,
625 MM_SETITEMATTR,
626 MPFROM2SHORT(lOldDefault,
627 FALSE),
628 MPFROM2SHORT(MIA_CHECKED, 0));
629 }
630
631 return brc;
632}

686/*
687 *@@ winhCopyMenuItem2:
688 * copies a menu item from hmenuSource to hmenuTarget.
689 *
690 * This creates a full duplicate. If usID specifies
691 * a submenu, the entire submenu is copied as well
692 * (this will then recurse).
693 *
694 * fl can be any combination of:
695 *
696 * -- COPYFL_STRIPTABS: strip off \t and everything
697 * that follows, if present.
698 *
699 * NOTE: Copying submenus will work only if each item
700 * in the submenu has a unique menu ID. This is due
701 * to the dumb implementation of menus in PM where
702 * it is impossible to query menu items without
703 * knowing their ID.
704 *
705 *@@added V0.9.9 (2001-03-09) [umoeller]
706 *@@changed V0.9.20 (2002-08-10) [umoeller]: renamed, added fl
707 */
708
709BOOL winhCopyMenuItem2(HWND hmenuTarget,
710 HWND hmenuSource,
711 USHORT usID,
712 SHORT sTargetPosition, // in: position to insert at or MIT_END
713 ULONG fl) // in: COPYFL_* flags
714{
715 BOOL brc = FALSE;
716 MENUITEM mi = {0};
717 if (WinSendMsg(hmenuSource,
718 MM_QUERYITEM,
719 MPFROM2SHORT(usID,
720 FALSE),
721 (MPARAM)&mi))
722 {
723 // found in source:
724 // is it a separator?
725 if (mi.afStyle & MIS_SEPARATOR)
726 winhInsertMenuSeparator(hmenuTarget,
727 sTargetPosition,
728 usID);
729 else
730 {
731 // no separator:
732 // get item text
733 PSZ pszSource;
734 if (pszSource = winhQueryMenuItemText(hmenuSource,
735 usID))
736 {
737 PSZ p;
738 // remove the hotkey description
739 // V0.9.20 (2002-08-10) [umoeller]
740 if ( (fl & COPYFL_STRIPTABS)
741 && (p = strchr(pszSource, '\t'))
742 )
743 *p = '\0';
744
745 if ( (mi.afStyle & MIS_SUBMENU)
746 && (mi.hwndSubMenu)
747 )
748 {
749 // this is the top of a submenu:
750 HWND hwndSubMenu;
751 if (hwndSubMenu = winhInsertSubmenu(hmenuTarget,
752 sTargetPosition,
753 mi.id,
754 pszSource,
755 mi.afStyle,
756 0,
757 NULL,
758 0,
759 0))
760 {
761 // now copy all the items in the submenu
762 SHORT cMenuItems = SHORT1FROMMR(WinSendMsg(mi.hwndSubMenu,
763 MM_QUERYITEMCOUNT,
764 0,
765 0));
766 // loop through all entries in the original submenu
767 ULONG i;
768 for (i = 0;
769 i < cMenuItems;
770 i++)
771 {
772 SHORT id = SHORT1FROMMR(WinSendMsg(mi.hwndSubMenu,
773 MM_ITEMIDFROMPOSITION,
774 MPFROMSHORT(i),
775 0));

1338 * give, say, every tenth item a taller tick.

1347 *

1357 *

1384 ulSet < 2;
1385 ulSet++)
1386 {

1388 {

1391 SLM_SETTICKSIZE,
1392 MPFROM2SHORT(SMA_SETALLTICKS,
1393 ulPixels),

1430 }

1899/*
1900 *@@ winhUpdateScrollBar:
1901 * updates the given scroll bar according to the given
1902 * values. This updates the scroll bar's thumb size,
1903 * extension, and position, all in one shot.
1904 *
1905 * This function usually gets called when the window is
1906 * created and later when the window is resized.
1907 *
1908 * This simplifies the typical functionality of a scroll
1909 * bar in a client window which is to be scrolled. I am
1910 * wondering why IBM never included such a function, since
1911 * it is so damn basic and still writing it cost me a whole
1912 * day.
1913 *
1914 * Terminology:
1915 *
1916 * -- "window": the actual window with scroll bars which displays
1917 * a subrectangle of the available data. With a typical PM
1918 * application, this will be your client window.
1919 *
1920 * The width or height of this must be passed in ulWinPels.
1921 *
1922 * -- "workarea": the entire data to be displayed, of which the
1923 * "window" can only display a subrectangle, if the workarea
1924 * is larger than the window.
1925 *
1926 * The width or height of this must be passed in ulWorkareaPels.
1927 * This can be smaller than ulWinPels (if the window is larger
1928 * than the data) or the same or larger than ulWinPels
1929 * (if the window is too small to show all the data).
1930 *
1931 * This value is exclusive in the sense that the maximum
1932 * window offset (below) can be the workarea minus one.
1933 *
1934 * -- "window offset": the zero-based offset of the current
1935 * window within the workarea, whose maximum value is
1936 * the workarea minus one.
1937 *
1938 * [pr]: I disagree with the above "workarea minus one" stuff.
1939 * It is perfectly possible to have the workarea and the
1940 * window area the same size and to display all the content.
1941 * This also means you do NOT get a disabled scroll bar
1942 * any more when they are the same size.
1943 *
1944 * For horizontal scroll bars, this is the X coordinate,
1945 * counting from the left of the window (0 means leftmost).
1946 *
1947 * For vertical scroll bars, this is counted from the _top_
1948 * of the workarea (0 means topmost, as opposed to OS/2
1949 * window coordinates!). This is because for vertical scroll
1950 * bars controls, higher values move the thumb _down_. Yes
1951 * indeed, this conflicts with PM's coordinate system.
1952 *
1953 * The window offset is therefore always positive.
1954 *
1955 * The scroll bar gets disabled if the entire workarea is visible,
1956 * that is, if ulWorkareaPels <= ulWinPels. In that case
1957 * FALSE is returned. If (fAutoHide == TRUE), the scroll
1958 * bar is not only disabled, but also hidden from the display.
1959 * In that case, you will need to reformat your output because
1960 * your workarea becomes larger without the scroll bar.
1961 *
1962 * This function will set the range of the scroll bar to 0 up
1963 * to a value depending on the workarea size. For vertical scroll
1964 * bars, 0 means topmost (which is kinda sick with the OS/2
1965 * coordinate system), for horizontal scroll bars, 0 means leftmost.
1966 *
1967 * The maximum value of the scroll bar will be
1968 *
1969 + (ulWorkareaPels - ulWinPels) / usScrollUnitPels
1970 *
1971 * The thumb size of the scroll bar will also be adjusted
1972 * based on the workarea and window size, as it should be.
1973 *
1974 *@@added V0.9.1 (2000-02-14) [umoeller]
1975 *@@changed V0.9.3 (2000-04-30) [umoeller]: fixed pels/unit confusion
1976 *@@changed V0.9.3 (2000-05-08) [umoeller]: now handling scroll units automatically
1977 *@@changed V1.0.1 (2003-01-25) [umoeller]: fixed max value which caused right/bottommost scroll button to never be disabled
1978 *@@changed V1.0.1 (2003-01-25) [umoeller]: fixed bad thumb position for large offsets
1979 *@@changed WarpIN V1.0.18 (2008-11-16) [pr]: fix rounding errors @@fixes 1086
1980 *@@changed WarpIN V1.0.18 (2008-11-16) [pr]: disable scroll bars when workarea = win area @@fixes 1086
1981 */
