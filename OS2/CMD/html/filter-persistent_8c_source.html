72static int _read_array(struct pfilter *pf, struct config_tree *cft,
73 const char *path, void *data)
74{
75 const struct config_node *cn;
76 struct config_value *cv;
77
78 if (!(cn = find_config_node(cft->root, path))) {
79 log_very_verbose("Couldn't find %s array in '%s'",
80 path, pf->file);
81 return 0;
82 }
83
84 /*
85 * iterate through the array, adding
86 * devices as we go.
87 */
88 for (cv = cn->v; cv; cv = cv->next) {
89 if (cv->type != CFG_STRING) {
90 log_verbose("Devices array contains a value "
91 "which is not a string ... ignoring");
92 continue;
93 }
94
95 if (!dm_hash_insert(pf->devices, cv->v.str, data))
96 log_verbose("Couldn't add '%s' to filter ... ignoring",
97 cv->v.str);
98 /* Populate dev_cache ourselves */
99 dev_cache_get(cv->v.str, NULL);
100 }
101 return 1;
102}

104int persistent_filter_load(struct dev_filter *f, struct config_tree **cft_out)
105{
106 struct pfilter *pf = (struct pfilter *) f->private;
107 struct config_tree *cft;
108 struct stat info;
109 int r = 0;
110
111 if (!stat(pf->file, &info))
112 pf->ctime = info.st_ctime;
113 else {
114 log_very_verbose("%s: stat failed: %s", pf->file,
115 strerror(errno));
116 return_0;
117 }
118
119 if (!(cft = create_config_tree(pf->file, 1)))
120 return_0;
121
122 if (!read_config_file(cft))
123 goto_out;
124
125 _read_array(pf, cft, "persistent_filter_cache/valid_devices",
126 PF_GOOD_DEVICE);
127 /* We don't gain anything by holding invalid devices */
128 /* _read_array(pf, cft, "persistent_filter_cache/invalid_devices",
129 PF_BAD_DEVICE); */
130
131 /* Did we find anything? */
132 if (dm_hash_get_num_entries(pf->devices)) {
133 /* We populated dev_cache ourselves */
134 dev_cache_scan(0);
135 r = 1;
136 }
137
138 log_very_verbose("Loaded persistent filter cache from %s", pf->file);
139
140 out:
141 if (r && cft_out)
142 *cft_out = cft;
143 else
144 destroy_config_tree(cft);
145 return r;
146}

148static void _write_array(struct pfilter *pf, FILE *fp, const char *path,
149 void *data)
150{
151 void *d;
152 int first = 1;
153 struct dm_hash_node *n;
154
155 for (n = dm_hash_get_first(pf->devices); n;
156 n = dm_hash_get_next(pf->devices, n)) {
157 d = dm_hash_get_data(pf->devices, n);
158
159 if (d != data)
160 continue;
161
162 if (!first)
163 fprintf(fp, ",\n");
164 else {
165 fprintf(fp, "\t%s=[\n", path);
166 first = 0;
167 }
168
169 fprintf(fp, "\t\t\"%s\"", dm_hash_get_key(pf->devices, n));
170 }
171
172 if (!first)
173 fprintf(fp, "\n\t]\n");
174
175 return;
176}

178int persistent_filter_dump(struct dev_filter *f)
179{
180 struct pfilter *pf = (struct pfilter *) f->private;
181 char *tmp_file;
182 struct stat info, info2;
183 struct config_tree *cft = NULL;
184 FILE *fp;
185 int lockfd;
186 int r = 0;
187
188 if (!dm_hash_get_num_entries(pf->devices)) {
189 log_very_verbose("Internal persistent device cache empty "
190 "- not writing to %s", pf->file);
191 return 0;
192 }
193 if (!dev_cache_has_scanned()) {
194 log_very_verbose("Device cache incomplete - not writing "
195 "to %s", pf->file);
196 return 0;
197 }
198
199 log_very_verbose("Dumping persistent device cache to %s", pf->file);
200
201 while (1) {
202 if ((lockfd = fcntl_lock_file(pf->file, F_WRLCK, 0)) < 0)
203 return_0;
204
205 /*
206 * Ensure we locked the file we expected
207 */
208 if (fstat(lockfd, &info)) {
209 log_sys_error("fstat", pf->file);
210 goto out;
211 }
212 if (stat(pf->file, &info2)) {
213 log_sys_error("stat", pf->file);
214 goto out;
215 }
216
217 if (!memcmp(&info.st_ino, &info2.st_ino, sizeof(ino_t)))
218 break;
219
220 fcntl_unlock_file(lockfd);
221 }
222
223 /*
224 * If file contents changed since we loaded it, merge new contents
225 */
226 if (info.st_ctime != pf->ctime)
227 /* Keep cft open to avoid losing lock */
228 persistent_filter_load(f, &cft);
229
230 tmp_file = alloca(strlen(pf->file) + 5);
231 sprintf(tmp_file, "%s.tmp", pf->file);
232
233 if (!(fp = fopen(tmp_file, "w"))) {
234 /* EACCES has been reported over NFS */
235 if (errno != EROFS && errno != EACCES)
236 log_sys_error("fopen", tmp_file);
237 goto out;
238 }
239
240 fprintf(fp, "# This file is automatically maintained by lvm.\n\n");
241 fprintf(fp, "persistent_filter_cache {\n");
242
243 _write_array(pf, fp, "valid_devices", PF_GOOD_DEVICE);
244 /* We don't gain anything by remembering invalid devices */
245 /* _write_array(pf, fp, "invalid_devices", PF_BAD_DEVICE); */
246
247 fprintf(fp, "}\n");
248 if (fclose(fp)) {
249 log_sys_error("fclose", tmp_file);
250 goto out;
251 }
252
253 if (rename(tmp_file, pf->file))
254 log_error("%s: rename to %s failed: %s", tmp_file, pf->file,
255 strerror(errno));
256
257 r = 1;
258
259out:
260 fcntl_unlock_file(lockfd);
261
262 if (cft)
263 destroy_config_tree(cft);
264
265 return r;
266}

268static int _lookup_p(struct dev_filter *f, struct device *dev)
269{
270 struct pfilter *pf = (struct pfilter *) f->private;
271 void *l = dm_hash_lookup(pf->devices, dev_name(dev));
272 struct str_list *sl;
273
274 if (!l) {
275 l = pf->real->passes_filter(pf->real, dev) ?
276 PF_GOOD_DEVICE : PF_BAD_DEVICE;
277
278 list_iterate_items(sl, struct str_list, &dev->aliases)
279 dm_hash_insert(pf->devices, sl->str, l);
280
281 } else if (l == PF_BAD_DEVICE)
282 log_debug("%s: Skipping (cached)", dev_name(dev));
283
284 return (l == PF_BAD_DEVICE) ? 0 : 1;
285}

298struct dev_filter *persistent_filter_create(struct dev_filter *real,
299 const char *file)
300{
301 struct pfilter *pf;
302 struct dev_filter *f = NULL;
303
304 if (!(pf = dm_malloc(sizeof(*pf)))) {
305 stack;
306 return NULL;
307 }
308 memset(pf, 0, sizeof(*pf));
309
310 if (!(pf->file = dm_malloc(strlen(file) + 1)))
311 goto_bad;
312
313 strcpy(pf->file, file);
314 pf->real = real;
315
316 if (!(_init_hash(pf))) {
317 log_error("Couldn't create hash table for persistent filter.");
318 goto bad;
319 }
320
321 if (!(f = dm_malloc(sizeof(*f))))
322 goto_bad;
323
324 f->passes_filter = _lookup_p;
325 f->destroy = _persistent_destroy;
326 f->private = pf;
327
328 return f;
329
330 bad:
331 dm_free(pf->file);
332 if (pf->devices)
333 dm_hash_destroy(pf->devices);
334 dm_free(pf);
335 dm_free(f);
336 return NULL;
337}
