33/* Bits set in the FLAGS argument to `fnmatch'. */
34#define FNM_PATHNAME (1 << 0)/* No wildcard can ever match `/'. */
35#define FNM_NOESCAPE (1 << 1)/* Backslashes don't quote special chars. */
36#define FNM_PERIOD (1 << 2)/* Leading `.' is matched only explicitly. */
37#define FNM_IGNORECASE (1 << 3)/* Ignore case when matching. */ /* MH */
38#define __FNM_FLAGS (FNM_PATHNAME|FNM_NOESCAPE|FNM_PERIOD|FNM_IGNORECASE) /* MH */
39
40/* Value returned by `fnmatch' if STRING does not match PATTERN. */
41#define FNM_NOMATCH 1
42
43/* Match STRING against the filename pattern PATTERN,
44 returning zero if it matches, FNM_NOMATCH if not. */
45extern int thematch();
