25FILE * LogFileInit( BOOL fAppend )
26{
27 FILE *pLog; // log file handle
28 PSZ pszLogPath, // path to store log file
29 pszLogFile; // filespec of log file
30 time_t ltime; // current time
31
32 if ( DosScanEnv( LOG_PATH, &pszLogPath ) == NO_ERROR ) {
33 pszLogFile = (PSZ) malloc( strlen( pszLogPath ) +
34 strlen( LOG_FILE ) + 2 );
35 sprintf( pszLogFile, "%s\\%s", pszLogPath, LOG_FILE );
36 pLog = fopen( pszLogFile, fAppend? "a": "w");
37 free( pszLogFile );
38 }
39 else pLog = fopen( LOG_FILE, fAppend? "a": "w");
40
41 if ( pLog ) {
42 time( &ltime );
43 fprintf( pLog, "*******************************************************************************\n");
44 fprintf( pLog, "LVMPM started: %s\n", ctime( &ltime ));
45 }
46 return ( pLog );
47}

50/* ------------------------------------------------------------------------- *
51 * Log_DiskInfo() *
52 * *
53 * Writes a summary of the disk layout to the log file (if logging is *
54 * enabled). *
55 * *
56 * ARGUMENTS: *
57 * PDVMGLOBAL pGlobal : Structure containing global program data *
58 * *
59 * RETURNS: N/A *
60 * ------------------------------------------------------------------------- */
61void Log_DiskInfo( PDVMGLOBAL pGlobal )
62{
63 Partition_Information_Array partitions;
64 ULONG i, j;
65 CARDINAL32 iErr;
66
67 if ( !pGlobal->pLog ) return;
68
69 fprintf( pGlobal->pLog, "-------------------------------------------------------------------------------\n");
70 fprintf( pGlobal->pLog, "LVM Summary: Physical\n" );
71 fprintf( pGlobal->pLog, "%u disk drives reported.\n", pGlobal->ulDisks );
72
73 for ( i = 0; i < pGlobal->ulDisks; i++ ) {
74 fprintf( pGlobal->pLog, "...............................................................................\n");
75 fprintf( pGlobal->pLog, "D%02u: \"%s\"\n",
76 pGlobal->disks[ i ].iNumber,
77 pGlobal->disks[ i ].szName );
78 fprintf( pGlobal->pLog, " Hnd: 0x%08X SlNo: %08X Unus: %u Corr: %u PRM: %u BFlp: %u\n",
79 pGlobal->disks[ i ].handle,
80 pGlobal->disks[ i ].iSerial,
81 pGlobal->disks[ i ].fUnusable,
82 pGlobal->disks[ i ].fCorrupt,
83 pGlobal->disks[ i ].fPRM,
84 pGlobal->disks[ i ].fBigFloppy );
85 fprintf( pGlobal->pLog, " Tot: %-11u Free: %-11u C: %-5u H: %-3u S: %-u\n",
86 MiB_TO_SECS( pGlobal->disks[ i ].iSize ),
87 pGlobal->disks[ i ].iTotalFree,
88 pGlobal->disks[ i ].iCylinders,
89 pGlobal->disks[ i ].iHeads,
90 pGlobal->disks[ i ].iSectors );
91
92 partitions = LvmGetPartitions( pGlobal->disks[ i ].handle, &iErr );
93 if ( iErr != LVM_ENGINE_NO_ERROR ) {
94 fprintf( pGlobal->pLog, "\n Unable to query partitions (error code %u)!\n\n", iErr );
95 continue;
96 }
97 for ( j = 0; j < partitions.Count; j++ ) {
98 Log_Partition( pGlobal, j+1, partitions.Partition_Array[ j ] );
99 }
100 LvmFreeMem( partitions.Partition_Array );
101// fprintf( pGlobal->pLog, "\n");
102 }
103}

108void Log_Partition( PDVMGLOBAL pGlobal, ULONG ulNum, Partition_Information_Record pir )
109{
110 if ( !pGlobal->pLog ) return;
111
112 fprintf( pGlobal->pLog, " . . . . . . . . . . . . . . . . . . . . . . . . . . . . . . . . . . . . .\n");
113 fprintf( pGlobal->pLog, " P%02u: \"%s\"", ulNum, pir.Partition_Name );
114 if ( pir.Volume_Handle )
115 fprintf( pGlobal->pLog, " (in volume \"%s\")", pir.Volume_Name );
116 fprintf( pGlobal->pLog, "\n");
117 fprintf( pGlobal->pLog, " Hnd: 0x%08X DrvH: 0x%08X",
118 pir.Partition_Handle, pir.Drive_Handle );
119 if ( pir.Volume_Handle )
120 fprintf( pGlobal->pLog, " VolH: 0x%08X", pir.Volume_Handle );
121 fprintf( pGlobal->pLog, "\n");
122 fprintf( pGlobal->pLog, " Sec: %-10u Len: %-10u Usbl: %-10u (%u MiB)\n",
123 pir.Partition_Start, pir.True_Partition_Size, pir.Usable_Partition_Size,
124 SECS_TO_MiB( pir.Usable_Partition_Size ));
125 fprintf( pGlobal->pLog, " Sta: %c Pri: %u BMgr: %u Act: %u OS: 0x%02X",
126 pir.Partition_Status ?
127 (pir.Partition_Status == PARTITION_IS_IN_USE? 'U': 'A') :
128 'F',
129 pir.Primary_Partition,
130 pir.On_Boot_Manager_Menu? 1: 0,
131 ( pir.Primary_Partition && ( pir.Active_Flag == ACTIVE_PARTITION ))? 1: 0,
132 pir.OS_Flag );
133 if ( strlen( pir.File_System_Name ))
134 fprintf( pGlobal->pLog, " (%s)", pir.File_System_Name );
135 fprintf( pGlobal->pLog, "\n");
136}

151{

153

155

157 fprintf( pGlobal->pLog, "LVM Summary: Logical\n" );
158 fprintf( pGlobal->pLog, "%u volumes reported.\n", pGlobal->ulVolumes );
159
160 for ( i = 0; i < pGlobal->ulVolumes; i++ ) {
161 fprintf( pGlobal->pLog, "...............................................................................\n");
162// fprintf( pGlobal->pLog, "\n");
163 if ( pGlobal->volumes[ i ].cLetter )
164 fprintf( pGlobal->pLog, "%c:", pGlobal->volumes[ i ].cLetter );
165 else
166 fprintf( pGlobal->pLog, " ", pGlobal->volumes[ i ].cLetter );
167 fprintf( pGlobal->pLog, " \"%s\" (%s)\n",
168 pGlobal->volumes[ i ].szName,
169 pGlobal->volumes[ i ].szFS );
170 fprintf( pGlobal->pLog, " Hnd: 0x%08X Dev: %u Pref: %c Init: %c Conf: 0x%X\n",
171 pGlobal->volumes[ i ].handle,
172 pGlobal->volumes[ i ].bDevice,
173 pGlobal->volumes[ i ].cPreference,
174 pGlobal->volumes[ i ].cInitial,
175 pGlobal->volumes[ i ].iConflict );
176 fprintf( pGlobal->pLog, " Com: %u Stat: %u New: %u Prts: %u Size: %u MiB\n",
177 pGlobal->volumes[ i ].fCompatibility,
178 pGlobal->volumes[ i ].iStatus,
179 pGlobal->volumes[ i ].fNew,
180 pGlobal->volumes[ i ].iPartitions,

187/* ------------------------------------------------------------------------- *
188 * Log_CreatePartition() *
189 * *
190 * ARGUMENTS: *
191 * PDVMGLOBAL pGlobal: Structure containing global program data *
192 * DVMCREATEPARMS data : Partition creation parameters *
193 * *
194 * RETURNS: N/A *

196void Log_CreatePartition( PDVMGLOBAL pGlobal, DVMCREATEPARMS data, ADDRESS handle, CARDINAL32 iRC )
197{
198 if ( !pGlobal->pLog ) return;
199
200 fprintf( pGlobal->pLog, "-------------------------------------------------------------------------------\n");
201 fprintf( pGlobal->pLog, "Creating partition:\n");
202 fprintf( pGlobal->pLog, "\"%s\" (%s)\n",
203 data.szName,
204 (data.fType & PARTITION_TYPE_PRIMARY)? "Primary": "Logical");
205 fprintf( pGlobal->pLog, "Size: %u MB (%s)\n", data.ulNumber,
206 (data.fType & PARTITION_FLAG_FROMEND)? "from end": "from start");
207 fprintf( pGlobal->pLog, "Free space: 0x%08X\n", data.pPartitions[ 0 ] );
208 fprintf( pGlobal->pLog, "New handle: 0x%08X\n", handle == NULL? 0: handle );
209 fprintf( pGlobal->pLog, "Result: %u\n", iRC );
210}
