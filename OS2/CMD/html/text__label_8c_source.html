40static int _text_write(struct label *label, void *buf)
41{
42 struct label_header *lh = (struct label_header *) buf;
43 struct pv_header *pvhdr;
44 struct lvmcache_info *info;
45 struct disk_locn *pvh_dlocn_xl;
46 struct metadata_area *mda;
47 struct mda_context *mdac;
48 struct data_area_list *da;
49
50 /* FIXME Move to where label is created */
51 strncpy(label->type, LVM2_LABEL, sizeof(label->type));
52
53 strncpy((char *)lh->type, label->type, sizeof(label->type));
54
55 pvhdr = (struct pv_header *) ((char *) buf + xlate32(lh->offset_xl));
56 info = (struct lvmcache_info *) label->info;
57 pvhdr->device_size_xl = xlate64(info->device_size);
58 memcpy(pvhdr->pv_uuid, &info->dev->pvid, sizeof(struct id));
59
60 pvh_dlocn_xl = &pvhdr->disk_areas_xl[0];
61
62 /* List of data areas (holding PEs) */
63 list_iterate_items(da, struct data_area_list, &info->das) {
64 pvh_dlocn_xl->offset = xlate64(da->disk_locn.offset);
65 pvh_dlocn_xl->size = xlate64(da->disk_locn.size);
66 pvh_dlocn_xl++;
67 }
68
69 /* NULL-termination */
70 pvh_dlocn_xl->offset = xlate64(UINT64_C(0));
71 pvh_dlocn_xl->size = xlate64(UINT64_C(0));
72 pvh_dlocn_xl++;
73
74 /* List of metadata area header locations */
75 list_iterate_items(mda, struct metadata_area, &info->mdas) {
76 mdac = (struct mda_context *) mda->metadata_locn;
77
78 if (mdac->area.dev != info->dev)
79 continue;
80
81 pvh_dlocn_xl->offset = xlate64(mdac->area.start);
82 pvh_dlocn_xl->size = xlate64(mdac->area.size);
83 pvh_dlocn_xl++;
84 }
85
86 /* NULL-termination */
87 pvh_dlocn_xl->offset = xlate64(UINT64_C(0));
88 pvh_dlocn_xl->size = xlate64(UINT64_C(0));
89
90 return 1;
91}

130int add_mda(const struct format_type *fmt, struct dm_pool *mem, struct list *mdas,
131 struct device *dev, uint64_t start, uint64_t size)
132{
133/* FIXME List size restricted by pv_header SECTOR_SIZE */
134 struct metadata_area *mdal;
135 struct mda_lists *mda_lists = (struct mda_lists *) fmt->private;
136 struct mda_context *mdac;
137
138 if (!mem) {
139 if (!(mdal = dm_malloc(sizeof(struct metadata_area)))) {
140 log_error("struct mda_list allocation failed");
141 return 0;
142 }
143
144 if (!(mdac = dm_malloc(sizeof(struct mda_context)))) {
145 log_error("struct mda_context allocation failed");
146 dm_free(mdal);
147 return 0;
148 }
149 } else {
150 if (!(mdal = dm_pool_alloc(mem, sizeof(struct metadata_area)))) {
151 log_error("struct mda_list allocation failed");
152 return 0;
153 }
154
155 if (!(mdac = dm_pool_alloc(mem, sizeof(struct mda_context)))) {
156 log_error("struct mda_context allocation failed");
157 return 0;
158 }
159 }
160
161 mdal->ops = mda_lists->raw_ops;
162 mdal->metadata_locn = mdac;
163
164 mdac->area.dev = dev;
165 mdac->area.start = start;
166 mdac->area.size = size;
167 memset(&mdac->rlocn, 0, sizeof(mdac->rlocn));
168
169 list_add(mdas, &mdal->list);
170 return 1;
171}

194static int _text_read(struct labeller *l, struct device *dev, void *buf,
195 struct label **label)
196{
197 struct label_header *lh = (struct label_header *) buf;
198 struct pv_header *pvhdr;
199 struct lvmcache_info *info;
200 struct disk_locn *dlocn_xl;
201 uint64_t offset;
202 struct metadata_area *mda;
203 struct id vgid;
204 struct mda_context *mdac;
205 const char *vgname;
206 uint32_t vgstatus;
207 char *creation_host;
208
209 pvhdr = (struct pv_header *) ((char *) buf + xlate32(lh->offset_xl));
210
211 if (!(info = lvmcache_add(l, (char *)pvhdr->pv_uuid, dev, NULL, NULL, 0)))
212 return_0;
213 *label = info->label;
214
215 info->device_size = xlate64(pvhdr->device_size_xl);
216
217 if (info->das.n)
218 del_das(&info->das);
219 list_init(&info->das);
220
221 if (info->mdas.n)
222 del_mdas(&info->mdas);
223 list_init(&info->mdas);
224
225 /* Data areas holding the PEs */
226 dlocn_xl = pvhdr->disk_areas_xl;
227 while ((offset = xlate64(dlocn_xl->offset))) {
228 add_da(NULL, &info->das, offset,
229 xlate64(dlocn_xl->size));
230 dlocn_xl++;
231 }
232
233 /* Metadata area headers */
234 dlocn_xl++;
235 while ((offset = xlate64(dlocn_xl->offset))) {
236 add_mda(info->fmt, NULL, &info->mdas, dev, offset,
237 xlate64(dlocn_xl->size));
238 dlocn_xl++;
239 }
240
241 list_iterate_items(mda, struct metadata_area, &info->mdas) {
242 mdac = (struct mda_context *) mda->metadata_locn;
243 if ((vgname = vgname_from_mda(info->fmt, &mdac->area,
244 &vgid, &vgstatus, &creation_host)) &&
245 !lvmcache_update_vgname_and_id(info, vgname,
246 (char *) &vgid, vgstatus,
247 creation_host))
248 return_0;
249 }
250
251 info->status &= ~CACHE_INVALID;
252
253 return 1;
254}
