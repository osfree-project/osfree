20static int _pvchange_single(struct cmd_context *cmd, struct physical_volume *pv,
21 void *handle) // __attribute((unused)))
22{
23 struct volume_group *vg = NULL;
24 struct pv_list *pvl;
25 struct list mdas;
26 uint64_t sector;
27 uint32_t orig_pe_alloc_count;
28
29 const char *pv_name = dev_name(pv_dev(pv));
30 const char *tag = NULL;
31 const char *orig_vg_name;
32 char uuid[64]; // __attribute((aligned(8)));
33
34 int consistent = 1;
35 int allocatable = 0;
36 int tagarg = 0;
37
38 list_init(&mdas);
39
40 if (arg_count(cmd, addtag_ARG))
41 tagarg = addtag_ARG;
42 else if (arg_count(cmd, deltag_ARG))
43 tagarg = deltag_ARG;
44
45 if (arg_count(cmd, allocatable_ARG))
46 allocatable = !strcmp(arg_str_value(cmd, allocatable_ARG, "n"),
47 "y");
48 else if (tagarg && !(tag = arg_str_value(cmd, tagarg, NULL))) {
49 log_error("Failed to get tag");
50 return 0;
51 }
52
53 /* If in a VG, must change using volume group. */
54 if (*pv_vg_name(pv)) {
55 log_verbose("Finding volume group of physical volume \"%s\"",
56 pv_name);
57
58 if (!lock_vol(cmd, pv_vg_name(pv), LCK_VG_WRITE)) {
59 log_error("Can't get lock for %s", pv_vg_name(pv));
60 return 0;
61 }
62
63 if (!(vg = vg_read(cmd, pv_vg_name(pv), NULL, &consistent))) {
64 unlock_vg(cmd, pv_vg_name(pv));
65 log_error("Unable to find volume group of \"%s\"",
66 pv_name);
67 return 0;
68 }
69
70 if (!vg_check_status(vg,
71 CLUSTERED | EXPORTED_VG | LVM_WRITE)) {
72 unlock_vg(cmd, pv_vg_name(pv));
73 return 0;
74 }
75
76 if (!(pvl = find_pv_in_vg(vg, pv_name))) {
77 unlock_vg(cmd, pv_vg_name(pv));
78 log_error
79 ("Unable to find \"%s\" in volume group \"%s\"",
80 pv_name, vg->name);
81 return 0;
82 }
83 if (tagarg && !(vg->fid->fmt->features & FMT_TAGS)) {
84 unlock_vg(cmd, pv_vg_name(pv));
85 log_error("Volume group containing %s does not "
86 "support tags", pv_name);
87 return 0;
88 }
89 if (arg_count(cmd, uuid_ARG) && lvs_in_vg_activated(vg)) {
90 unlock_vg(cmd, pv_vg_name(pv));
91 log_error("Volume group containing %s has active "
92 "logical volumes", pv_name);
93 return 0;
94 }
95 pv = pvl->pv;
96 if (!archive(vg))
97 return 0;
98 } else {
99 if (tagarg) {
100 log_error("Can't change tag on Physical Volume %s not "
101 "in volume group", pv_name);
102 return 0;
103 }
104 if (!lock_vol(cmd, ORPHAN, LCK_VG_WRITE)) {
105 log_error("Can't get lock for orphans");
106 return 0;
107 }
108
109 if (!(pv = pv_read(cmd, pv_name, &mdas, &sector, 1))) {
110 unlock_vg(cmd, ORPHAN);
111 log_error("Unable to read PV \"%s\"", pv_name);
112 return 0;
113 }
114
115 }
116
117 if (arg_count(cmd, allocatable_ARG)) {
118 if (!*pv_vg_name(pv) &&
119 !(pv->fmt->features & FMT_ORPHAN_ALLOCATABLE)) {
120 log_error("Allocatability not supported by orphan "
121 "%s format PV %s", pv->fmt->name, pv_name);
122 unlock_vg(cmd, ORPHAN);
123 return 0;
124 }
125
126 /* change allocatability for a PV */
127 if (allocatable && (pv_status(pv) & ALLOCATABLE_PV)) {
128 log_error("Physical volume \"%s\" is already "
129 "allocatable", pv_name);
130 if (*pv_vg_name(pv))
131 unlock_vg(cmd, pv_vg_name(pv));
132 else
133 unlock_vg(cmd, ORPHAN);
134 return 1;
135 }
136
137 if (!allocatable && !(pv_status(pv) & ALLOCATABLE_PV)) {
138 log_error("Physical volume \"%s\" is already "
139 "unallocatable", pv_name);
140 if (*pv_vg_name(pv))
141 unlock_vg(cmd, pv_vg_name(pv));
142 else
143 unlock_vg(cmd, ORPHAN);
144 return 1;
145 }
146
147 if (allocatable) {
148 log_verbose("Setting physical volume \"%s\" "
149 "allocatable", pv_name);
150 pv->status |= ALLOCATABLE_PV;
151 } else {
152 log_verbose("Setting physical volume \"%s\" NOT "
153 "allocatable", pv_name);
154 pv->status &= ~ALLOCATABLE_PV;
155 }
156 } else if (tagarg) {
157 /* tag or deltag */
158 if ((tagarg == addtag_ARG)) {
159 if (!str_list_add(cmd->mem, &pv->tags, tag)) {
160 log_error("Failed to add tag %s to physical "
161 "volume %s", tag, pv_name);
162 return 0;
163 }
164 } else {
165 if (!str_list_del(&pv->tags, tag)) {
166 log_error("Failed to remove tag %s from "
167 "physical volume" "%s", tag, pv_name);
168 return 0;
169 }
170 }
171 } else {
172 /* --uuid: Change PV ID randomly */
173 if (!id_create(&pv->id)) {
174 log_error("Failed to generate new random UUID for %s.",
175 pv_name);
176 return 0;
177 }
178 if (!id_write_format(&pv->id, uuid, sizeof(uuid))) {
179 stack;
180 return 0;
181 }
182 log_verbose("Changing uuid of %s to %s.", pv_name, uuid);
183 if (*pv_vg_name(pv)) {
184 orig_vg_name = pv_vg_name(pv);
185 orig_pe_alloc_count = pv_pe_alloc_count(pv);
186 pv->vg_name = ORPHAN;
187 pv->pe_alloc_count = 0;
188 if (!(pv_write(cmd, pv, NULL, INT64_C(-1)))) {
189 log_error("pv_write with new uuid failed "
190 "for %s.", pv_name);
191 return 0;
192 }
193 pv->vg_name = orig_vg_name;
194 pv->pe_alloc_count = orig_pe_alloc_count;
195 }
196 }
197
198 log_verbose("Updating physical volume \"%s\"", pv_name);
199 if (*pv_vg_name(pv)) {
200 if (!vg_write(vg) || !vg_commit(vg)) {
201 unlock_vg(cmd, pv_vg_name(pv));
202 log_error("Failed to store physical volume \"%s\" in "
203 "volume group \"%s\"", pv_name, vg->name);
204 return 0;
205 }
206 backup(vg);
207 unlock_vg(cmd, pv_vg_name(pv));
208 } else {
209 if (!(pv_write(cmd, pv, NULL, INT64_C(-1)))) {
210 unlock_vg(cmd, ORPHAN);
211 log_error("Failed to store physical volume \"%s\"",
212 pv_name);
213 return 0;
214 }
215 unlock_vg(cmd, ORPHAN);
216 }
217
218 log_print("Physical volume \"%s\" changed", pv_name);
219
220 return 1;
221}

223int pvchange(struct cmd_context *cmd, int argc, char **argv)
224{
225 int opt = 0;
226 int done = 0;
227 int total = 0;
228
229 struct physical_volume *pv;
230 char *pv_name;
231
232 struct pv_list *pvl;
233 struct list *pvslist;
234 struct list mdas;
235
236 list_init(&mdas);
237
238 if (arg_count(cmd, allocatable_ARG) + arg_count(cmd, addtag_ARG) +
239 arg_count(cmd, deltag_ARG) + arg_count(cmd, uuid_ARG) != 1) {
240 log_error("Please give exactly one option of -x, -uuid, "
241 "--addtag or --deltag");
242 return EINVALID_CMD_LINE;
243 }
244
245 if (!(arg_count(cmd, all_ARG)) && !argc) {
246 log_error("Please give a physical volume path");
247 return EINVALID_CMD_LINE;
248 }
249
250 if (arg_count(cmd, all_ARG) && argc) {
251 log_error("Option a and PhysicalVolumePath are exclusive");
252 return EINVALID_CMD_LINE;
253 }
254
255 if (argc) {
256 log_verbose("Using physical volume(s) on command line");
257 for (; opt < argc; opt++) {
258 pv_name = argv[opt];
259 /* FIXME Read VG instead - pv_read will fail */
260 if (!(pv = pv_read(cmd, pv_name, &mdas, NULL, 1))) {
261 log_error("Failed to read physical volume %s",
262 pv_name);
263 continue;
264 }
265 total++;
266 done += _pvchange_single(cmd, pv, NULL);
267 }
268 } else {
269 log_verbose("Scanning for physical volume names");
270 if (!(pvslist = get_pvs(cmd))) {
271 return ECMD_FAILED;
272 }
273
274 list_iterate_items(pvl, struct pv_list, pvslist) {
275 total++;
276 done += _pvchange_single(cmd, pvl->pv, NULL);
277 }
278 }
279
280 log_print("%d physical volume%s changed / %d physical volume%s "
281 "not changed",
282 done, done == 1 ? "" : "s",
283 total - done, (total - done) == 1 ? "" : "s");
284
285 return (total == done) ? ECMD_PROCESSED : ECMD_FAILED;
286}
