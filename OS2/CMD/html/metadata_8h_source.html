150struct metadata_area_ops {
151 struct volume_group *(*vg_read) (struct format_instance * fi,
152 const char *vg_name,
153 struct metadata_area * mda);
154 struct volume_group *(*vg_read_precommit) (struct format_instance * fi,
155 const char *vg_name,
156 struct metadata_area * mda);
157 /*
158 * Write out complete VG metadata. You must ensure internal
159 * consistency before calling. eg. PEs can't refer to PVs not
160 * part of the VG.
161 *
162 * It is also the responsibility of the caller to ensure external
163 * consistency, eg by calling pv_write() if removing PVs from
164 * a VG or calling vg_write() a second time if splitting a VG
165 * into two.
166 *
167 * vg_write() should not read or write from any PVs not included
168 * in the volume_group structure it is handed.
169 * (format1 currently breaks this rule.)
170 */
171 int (*vg_write) (struct format_instance * fid, struct volume_group * vg,
172 struct metadata_area * mda);
173 int (*vg_precommit) (struct format_instance * fid,
174 struct volume_group * vg,
175 struct metadata_area * mda);
176 int (*vg_commit) (struct format_instance * fid,
177 struct volume_group * vg, struct metadata_area * mda);
178 int (*vg_revert) (struct format_instance * fid,
179 struct volume_group * vg, struct metadata_area * mda);
180 int (*vg_remove) (struct format_instance * fi, struct volume_group * vg,
181 struct metadata_area * mda);
182 /*
183 * Check if metadata area belongs to vg
184 */
185 int (*mda_in_vg) (struct format_instance * fi,
186 struct volume_group * vg, struct metadata_area *mda);
187 /*
188 * Analyze a metadata area on a PV.
189 */
190 int (*pv_analyze_mda) (const struct format_type * fmt,
191 struct metadata_area *mda);
