48/* Open connection to the Cluster Manager daemon */
49static int _open_local_sock(void)
50{
51 int local_socket;
52 struct sockaddr_un sockaddr;
53
54 /* Open local socket */
55 if ((local_socket = socket(PF_UNIX, SOCK_STREAM, 0)) < 0) {
56 fprintf(stderr, "Local socket creation failed: %s", strerror(errno));
57 return -1;
58 }
59
60 memset(&sockaddr, 0, sizeof(sockaddr));
61 memcpy(sockaddr.sun_path, CLVMD_SOCKNAME, sizeof(CLVMD_SOCKNAME));
62
63 sockaddr.sun_family = AF_UNIX;
64
65 if (connect(local_socket,(struct sockaddr *) &sockaddr,
66 sizeof(sockaddr))) {
67 int saved_errno = errno;
68
69 fprintf(stderr, "connect() failed on local socket: %s\n",
70 strerror(errno));
71 if (close(local_socket))
72 return -1;
73
74 errno = saved_errno;
75 return -1;
76 }
77
78 return local_socket;
79}

81/* Send a request and return the status */
82static int _send_request(char *inbuf, int inlen, char **retbuf)
83{
84 char outbuf[PIPE_BUF];
85 struct clvm_header *outheader = (struct clvm_header *) outbuf;
86 int len;
87 int off;
88 int buflen;
89 int err;
90
91 /* Send it to CLVMD */
92 rewrite:
93 if ( (err = write(_clvmd_sock, inbuf, inlen)) != inlen) {
94 if (err == -1 && errno == EINTR)
95 goto rewrite;
96 fprintf(stderr, "Error writing data to clvmd: %s", strerror(errno));
97 return 0;
98 }
99
100 /* Get the response */
101 reread:
102 if ((len = read(_clvmd_sock, outbuf, sizeof(struct clvm_header))) < 0) {
103 if (errno == EINTR)
104 goto reread;
105 fprintf(stderr, "Error reading data from clvmd: %s", strerror(errno));
106 return 0;
107 }
108
109 if (len == 0) {
110 fprintf(stderr, "EOF reading CLVMD");
111 errno = ENOTCONN;
112 return 0;
113 }
114
115 /* Allocate buffer */
116 buflen = len + outheader->arglen;
117 *retbuf = dm_malloc(buflen);
118 if (!*retbuf) {
119 errno = ENOMEM;
120 return 0;
121 }
122
123 /* Copy the header */
124 memcpy(*retbuf, outbuf, len);
125 outheader = (struct clvm_header *) *retbuf;
126
127 /* Read the returned values */
128 off = 1; /* we've already read the first byte */
129 while (off <= outheader->arglen && len > 0) {
130 len = read(_clvmd_sock, outheader->args + off,
131 buflen - off - offsetof(struct clvm_header, args));
132 if (len > 0)
133 off += len;
134 }
135
136 /* Was it an error ? */
137 if (outheader->status != 0) {
138 errno = outheader->status;
139
140 /* Only return an error here if there are no node-specific
141 errors present in the message that might have more detail */
142 if (!(outheader->flags & CLVMD_FLAG_NODEERRS)) {
143 fprintf(stderr, "cluster request failed: %s\n", strerror(errno));
144 return 0;
145 }
146
147 }
148
149 return 1;
150}

152/* Build the structure header and parse-out wildcard node names */
153static void _build_header(struct clvm_header *head, int cmd, const char *node,
154 int len)
155{
156 head->cmd = cmd;
157 head->status = 0;
158 head->flags = 0;
159 head->clientid = 0;
160 head->arglen = len;
161
162 if (node) {
163 /*
164 * Allow a couple of special node names:
165 * "*" for all nodes,
166 * "." for the local node only
167 */
168 if (strcmp(node, "*") == 0) {
169 head->node[0] = '\0';
170 } else if (strcmp(node, ".") == 0) {
171 head->node[0] = '\0';
172 head->flags = CLVMD_FLAG_LOCAL;
173 } else
174 strcpy(head->node, node);
175 } else
176 head->node[0] = '\0';
177}

179/*
180 * Send a message to a(or all) node(s) in the cluster and wait for replies
181 */
182static int _cluster_request(char cmd, const char *node, void *data, int len,
183 lvm_response_t ** response, int *num)
184{
185 char outbuf[sizeof(struct clvm_header) + len + strlen(node) + 1];
186 char *inptr;
187 char *retbuf = NULL;
188 int status;
189 int i;
190 int num_responses = 0;
191 struct clvm_header *head = (struct clvm_header *) outbuf;
192 lvm_response_t *rarray;
193
194 *num = 0;
195
196 if (_clvmd_sock == -1)
197 _clvmd_sock = _open_local_sock();
198
199 if (_clvmd_sock == -1)
200 return 0;
201
202 _build_header(head, cmd, node, len);
203 memcpy(head->node + strlen(head->node) + 1, data, len);
204
205 status = _send_request(outbuf, sizeof(struct clvm_header) +
206 strlen(head->node) + len, &retbuf);
207 if (!status)
208 goto out;
209
210 /* Count the number of responses we got */
211 head = (struct clvm_header *) retbuf;
212 inptr = head->args;
213 while (inptr[0]) {
214 num_responses++;
215 inptr += strlen(inptr) + 1;
216 inptr += sizeof(int);
217 inptr += strlen(inptr) + 1;
218 }
219
220 /*
221 * Allocate response array.
222 * With an extra pair of INTs on the front to sanity
223 * check the pointer when we are given it back to free
224 */
225 *response = dm_malloc(sizeof(lvm_response_t) * num_responses +
226 sizeof(int) * 2);
227 if (!*response) {
228 errno = ENOMEM;
229 status = 0;
230 goto out;
231 }
232
233 rarray = *response;
234
235 /* Unpack the response into an lvm_response_t array */
236 inptr = head->args;
237 i = 0;
238 while (inptr[0]) {
239 strcpy(rarray[i].node, inptr);
240 inptr += strlen(inptr) + 1;
241
242 memcpy(&rarray[i].status, inptr, sizeof(int));
243 inptr += sizeof(int);
244
245 rarray[i].response = dm_malloc(strlen(inptr) + 1);
246 if (rarray[i].response == NULL) {
247 /* Free up everything else and return error */
248 int j;
249 for (j = 0; j < i; j++)
250 dm_free(rarray[i].response);
251 free(*response);
252 errno = ENOMEM;
253 status = -1;
254 goto out;
255 }
256
257 strcpy(rarray[i].response, inptr);
258 rarray[i].len = strlen(inptr);
259 inptr += strlen(inptr) + 1;
260 i++;
261 }
262 *num = num_responses;
263 *response = rarray;
264
265 out:
266 if (retbuf)
267 dm_free(retbuf);
268
269 return status;
270}

272/* Free reply array */
273static int _cluster_free_request(lvm_response_t * response, int num)
274{
275 int i;
276
277 for (i = 0; i < num; i++) {
278 dm_free(response[i].response);
279 }
280
281 dm_free(response);
282
283 return 1;

286int refresh_clvmd()
287{
288 int num_responses;
289 char args[1]; // No args really.
290 lvm_response_t *response;
291 int saved_errno;
292 int status;
293 int i;
294
295 status = _cluster_request(CLVMD_CMD_REFRESH, "*", args, 0, &response, &num_responses);
296
297 /* If any nodes were down then display them and return an error */
298 for (i = 0; i < num_responses; i++) {
299 if (response[i].status == EHOSTDOWN) {
300 fprintf(stderr, "clvmd not running on node %s",
301 response[i].node);
302 status = 0;
303 errno = response[i].status;
304 } else if (response[i].status) {
305 fprintf(stderr, "Error resetting node %s: %s",
306 response[i].node,
307 response[i].response[0] ?
308 response[i].response :
309 strerror(response[i].status));
310 status = 0;
311 errno = response[i].status;
312 }
313 }
314
315 saved_errno = errno;
316 _cluster_free_request(response, num_responses);
317 errno = saved_errno;
318
319 return status;
320}
