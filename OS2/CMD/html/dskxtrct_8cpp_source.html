91int ReplaceFilePrompt(char *FileName)
92{
93 char ReplyStr[MAXSTRLEN];
94 int Reply = -1;
95
96 while (Reply < 0)
97 {
98 cout << "Replace file: " << FileName << "?" << endl;
99 cout << " Select: (r)eplace, (s)kip, replace (a)ll, replace (n)one, or (f)ail" << endl;
100 cin >> ReplyStr;
101 if (stricmp(ReplyStr, "r") == 0)
102 Reply = CheckFileReplace;
103 if (stricmp(ReplyStr, "s") == 0)
104 Reply = CheckFileSkip;
105 if (stricmp(ReplyStr, "a") == 0)
106 {
107 Ctl.Replace = TRUE;
108 Reply = CheckFileReplace;
109 }
110 if (stricmp(ReplyStr, "n") == 0)
111 {
112 Ctl.ReplaceNone = TRUE;
113 Reply = CheckFileSkip;
114 }
115 if (stricmp(ReplyStr, "f") == 0)
116 {
117 PrintMsg(MsgFail, "Processing aborted by request");
118 exit(Exit_Fail);
119 }
120 }
121 return (Reply);
122}

131 return (ReplaceFilePrompt(FileName));
132}

166void BuildFile(DskFileCtl * FileCB)
167{
168 char xMsg[MAXSTRLEN], fname[MAXSTRLEN];
169 int OFile = 0, Create = FALSE;
170 struct stat StatBuf;
171
172 if (Ctl.CrcOnly)
173 return;
174 if (Ctl.ViewList)
175 {
176 BuildFilePrintMsg(MsgMust, FileCB, "F");
177 Ctl.CreateCount++;
178 return;
179 }
180
181 DskBuildName(fname, Ctl.Target, FileCB->pFullName);
182 int FileAccessReply = CheckFileAccess(fname);
183
184 if (Ctl.MsgLevel >= MsgFile)
185 BuildFilePrintMsgByType(FileAccessReply, FileCB);
186 if (FileAccessReply == CheckFileSkip)
187 {
188 Ctl.SkipCount++;
189 return;
190 }
191
192 if (-1 == (OFile = open(fname, O_CREAT | O_WRONLY | O_BINARY, S_IWRITE)))
193 {
194 sprintf(xMsg, "Unable to open output file: '%s'", fname);
195 PrintMsg(MsgError, xMsg);
196 }
197 else
198 {
199 Create = TRUE;
200 while (FileCB->DataSize > 0 && Create == TRUE)
201 {
202 if (write(OFile, FileCB->pData, FileCB->DataSize) != FileCB->DataSize)
203 {
204 sprintf(xMsg, "Error writing output file: %s", fname);
205 PrintMsg(MsgError, xMsg);
206 Create = FALSE;
207 }
208 else
209 DskReadNext(FileCB);
210 }
211 }
212 if (OFile != 0)
213 {
214 if (_chsize(OFile, FileCB->Size) < 0)
215 {
216 sprintf(xMsg, "Error setting size of target file: %s", fname);
217 PrintMsg(MsgError, xMsg);
218 Create = FALSE;
219 }
220 if (close(OFile) < 0)
221 {
222 sprintf(xMsg, "Error closing input file: %s", fname);
223 PrintMsg(MsgError, xMsg);
224 Create = FALSE;
225 }
226 struct utimbuf ubuf;
227
228 if (utime(fname, DskBuildDateTime(&ubuf, FileCB)) < 0)
229 {
230 sprintf(xMsg, "Error setting date/time of target file: %s", fname);
231 PrintMsg(MsgError, xMsg);
232 Create = FALSE;
233 }
234 }
235 if (!Create)
236 Ctl.ErrorCount++;
237 else if (FileAccessReply == CheckFileCreate)
238 Ctl.CreateCount++;
239 else
240 Ctl.ReplaceCount++;
241}

243void ProcessDskFile(char *sfile)
244{
245 char xMsg[MAXSTRLEN];
246 DskFileCtl *pFileCB = 0;
247
248 sprintf(xMsg, "Processing input file: '%s'", sfile);
249 PrintMsg(MsgStd, xMsg);
250 if (access(sfile, 0) != 0)
251 ErrorExit1("Unable to access source dsk file: %s", sfile);
252 int rc = DskFindFirst(sfile, &pFileCB);
253
254 if (rc != Dsk_no_more_files && Ctl.CommentDisplay == TRUE)
255 PrintMsg(MsgInfo, pFileCB->pDskCB->pBuf + 40); // print file comments
256 while (rc != Dsk_no_more_files)
257 {
258 if (pFileCB->Attributes & DskDirAttrSubDir)
259 BuildTargetDirectory(pFileCB->pFullName);
260 else
261 BuildFile(pFileCB);
262 rc = DskFileNext(pFileCB);
263 }
264 DskClose(pFileCB);
265}

268{
269 int found = FALSE;

272 for (int i = strlen(sdir); i > 0; i--)
273 {
274 if (sdir[i] == '\\')
275 {
276 sdir[i] = '\0';
277 found = TRUE;
278 break;
279 }
280 }
281 if (found == FALSE)
282 strcpy(sdir, "");

286void ProcessInputMask()
287{
288 HDIR SDirHandle = HDIR_SYSTEM;
289 FILEFINDBUF3 FindBuffer = {0};
290 ULONG ResultBufLen = sizeof(FILEFINDBUF3);
291 ULONG FindCount = 1;
292 APIRET rc = NO_ERROR;
293 char Sdir[MAXSTRLEN];
294 char fname[MAXSTRLEN];
295
296 BuildSDir(Sdir, Ctl.Source);
297 rc = DosFindFirst(Ctl.Source, &SDirHandle, FILE_NORMAL, &FindBuffer, ResultBufLen,
298 &FindCount, FIL_STANDARD);
299 if (rc != NO_ERROR)
300 ErrorExit1("Search for input files failed: %s", Ctl.Source);
301 while (rc != ERROR_NO_MORE_FILES)
302 {
303 Ctl.DskCount++;
304 DskBuildName(fname, Sdir, FindBuffer.achName);
305 ProcessDskFile(fname);
306 FindCount = 1;
307 rc = DosFindNext(SDirHandle, &FindBuffer, ResultBufLen, &FindCount);
308 if (rc != NO_ERROR && rc != ERROR_NO_MORE_FILES)
309 ErrorExit1("Search for additional input files failed: %s", Ctl.Source);
310 }
311 rc = DosFindClose(SDirHandle);
312}
