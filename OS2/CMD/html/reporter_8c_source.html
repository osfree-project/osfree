58static int _pvsegs_sub_single(struct cmd_context *cmd, struct volume_group *vg,
59 struct pv_segment *pvseg, void *handle)
60{
61 int consistent = 0;
62 struct physical_volume *pv = pvseg->pv;
63 int ret = ECMD_PROCESSED;
64
65 if (!lock_vol(cmd, pv_vg_name(pv), LCK_VG_READ)) {
66 log_error("Can't lock %s: skipping", pv_vg_name(pv));
67 return ECMD_FAILED;
68 }
69
70 if (!(vg = vg_read(cmd, pv_vg_name(pv), NULL, &consistent))) {
71 log_error("Can't read %s: skipping", pv_vg_name(pv));
72 goto out;
73 }
74
75 if (!vg_check_status(vg, CLUSTERED)) {
76 ret = ECMD_FAILED;
77 goto out;
78 }
79
80 if (!report_object(handle, vg, NULL, pv, NULL, pvseg))
81 ret = ECMD_FAILED;
82
83out:
84 unlock_vg(cmd, pv_vg_name(pv));
85 return ret;
86}

104static int _pvs_single(struct cmd_context *cmd, struct volume_group *vg,
105 struct physical_volume *pv, void *handle)
106{
107 int consistent = 0;
108 int ret = ECMD_PROCESSED;
109
110 if (pv_vg_name(pv)) {
111 if (!lock_vol(cmd, pv_vg_name(pv), LCK_VG_READ)) {
112 log_error("Can't lock %s: skipping", pv_vg_name(pv));
113 return ECMD_FAILED;
114 }
115
116 if (!(vg = vg_read(cmd, pv_vg_name(pv), (char *)&pv->vgid, &consistent))) {
117 log_error("Can't read %s: skipping", pv_vg_name(pv));
118 goto out;
119 }
120
121 if (!vg_check_status(vg, CLUSTERED)) {
122 ret = ECMD_FAILED;
123 goto out;
124 }
125 }
126
127 if (!report_object(handle, vg, NULL, pv, NULL, NULL))
128 ret = ECMD_FAILED;
129
130out:
131 if (pv_vg_name(pv))
132 unlock_vg(cmd, pv_vg_name(pv));
133
134 return ret;
135}

161static int _report(struct cmd_context *cmd, int argc, char **argv,
162 report_type_t report_type)
163{
164 void *report_handle;
165 const char *opts;
166 char *str;
167 const char *keys = NULL, *options = NULL, *separator;
168 int r = ECMD_PROCESSED;
169 int aligned, buffered, headings;
170 unsigned args_are_pvs;
171
172 aligned = find_config_tree_int(cmd, "report/aligned",
173 DEFAULT_REP_ALIGNED);
174 buffered = find_config_tree_int(cmd, "report/buffered",
175 DEFAULT_REP_BUFFERED);
176 headings = find_config_tree_int(cmd, "report/headings",
177 DEFAULT_REP_HEADINGS);
178 separator = find_config_tree_str(cmd, "report/separator",
179 DEFAULT_REP_SEPARATOR);
180
181 args_are_pvs = (report_type == PVS || report_type == PVSEGS) ? 1 : 0;
182
183 switch (report_type) {
184 case LVS:
185 keys = find_config_tree_str(cmd, "report/lvs_sort",
186 DEFAULT_LVS_SORT);
187 if (!arg_count(cmd, verbose_ARG))
188 options = find_config_tree_str(cmd,
189 "report/lvs_cols",
190 DEFAULT_LVS_COLS);
191 else
192 options = find_config_tree_str(cmd,
193 "report/lvs_cols_verbose",
194 DEFAULT_LVS_COLS_VERB);
195 break;
196 case VGS:
197 keys = find_config_tree_str(cmd, "report/vgs_sort",
198 DEFAULT_VGS_SORT);
199 if (!arg_count(cmd, verbose_ARG))
200 options = find_config_tree_str(cmd,
201 "report/vgs_cols",
202 DEFAULT_VGS_COLS);
203 else
204 options = find_config_tree_str(cmd,
205 "report/vgs_cols_verbose",
206 DEFAULT_VGS_COLS_VERB);
207 break;
208 case PVS:
209 keys = find_config_tree_str(cmd, "report/pvs_sort",
210 DEFAULT_PVS_SORT);
211 if (!arg_count(cmd, verbose_ARG))
212 options = find_config_tree_str(cmd,
213 "report/pvs_cols",
214 DEFAULT_PVS_COLS);
215 else
216 options = find_config_tree_str(cmd,
217 "report/pvs_cols_verbose",
218 DEFAULT_PVS_COLS_VERB);
219 break;
220 case SEGS:
221 keys = find_config_tree_str(cmd, "report/segs_sort",
222 DEFAULT_SEGS_SORT);
223 if (!arg_count(cmd, verbose_ARG))
224 options = find_config_tree_str(cmd,
225 "report/segs_cols",
226 DEFAULT_SEGS_COLS);
227 else
228 options = find_config_tree_str(cmd,
229 "report/segs_cols_verbose",
230 DEFAULT_SEGS_COLS_VERB);
231 break;
232 case PVSEGS:
233 keys = find_config_tree_str(cmd, "report/pvsegs_sort",
234 DEFAULT_PVSEGS_SORT);
235 if (!arg_count(cmd, verbose_ARG))
236 options = find_config_tree_str(cmd,
237 "report/pvsegs_cols",
238 DEFAULT_PVSEGS_COLS);
239 else
240 options = find_config_tree_str(cmd,
241 "report/pvsegs_cols_verbose",
242 DEFAULT_PVSEGS_COLS_VERB);
243 break;
244 }
245
246 /* If -o supplied use it, else use default for report_type */
247 if (arg_count(cmd, options_ARG)) {
248 opts = arg_str_value(cmd, options_ARG, "");
249 if (!opts || !*opts) {
250 log_error("Invalid options string: %s", opts);
251 return 0;
252 }
253 if (*opts == '+') {
254 if (!(str = dm_pool_alloc(cmd->mem,
255 strlen(options) + strlen(opts) + 1))) {
256 log_error("options string allocation failed");
257 return 0;
258 }
259 strcpy(str, options);
260 strcat(str, ",");
261 strcat(str, opts + 1);
262 options = str;
263 } else
264 options = opts;
265 }
266
267 /* -O overrides default sort settings */
268 if (arg_count(cmd, sort_ARG))
269 keys = arg_str_value(cmd, sort_ARG, "");
270
271 if (arg_count(cmd, separator_ARG))
272 separator = arg_str_value(cmd, separator_ARG, " ");
273 if (arg_count(cmd, separator_ARG))
274 aligned = 0;
275 if (arg_count(cmd, aligned_ARG))
276 aligned = 1;
277 if (arg_count(cmd, unbuffered_ARG) && !arg_count(cmd, sort_ARG))
278 buffered = 0;
279 if (arg_count(cmd, noheadings_ARG))
280 headings = 0;
281
282 if (!(report_handle = report_init(cmd, options, keys, &report_type,
283 separator, aligned, buffered,
284 headings)))
285 return_0;
286
287 /* Ensure options selected are compatible */
288 if (report_type & SEGS)
289 report_type |= LVS;
290 if (report_type & PVSEGS)
291 report_type |= PVS;
292 if ((report_type & LVS) && (report_type & PVS)) {
293 log_error("Can't report LV and PV fields at the same time");
294 dm_report_free(report_handle);
295 return 0;
296 }
297
298 /* Change report type if fields specified makes this necessary */
299 if (report_type & SEGS)
300 report_type = SEGS;
301 else if (report_type & LVS)
302 report_type = LVS;
303 else if (report_type & PVSEGS)
304 report_type = PVSEGS;
305 else if (report_type & PVS)
306 report_type = PVS;
307
308 switch (report_type) {
309 case LVS:
310 r = process_each_lv(cmd, argc, argv, LCK_VG_READ, report_handle,
311 &_lvs_single);
312 break;
313 case VGS:
314 r = process_each_vg(cmd, argc, argv, LCK_VG_READ, 0,
315 report_handle, &_vgs_single);
316 break;
317 case PVS:
318 if (args_are_pvs)
319 r = process_each_pv(cmd, argc, argv, NULL,
320 report_handle, &_pvs_single);
321 else
322 r = process_each_vg(cmd, argc, argv, LCK_VG_READ, 0,
323 report_handle, &_pvs_in_vg);
324 break;
325 case SEGS:
326 r = process_each_lv(cmd, argc, argv, LCK_VG_READ, report_handle,
327 &_lvsegs_single);
328 break;
329 case PVSEGS:
330 if (args_are_pvs)
331 r = process_each_pv(cmd, argc, argv, NULL,
332 report_handle, &_pvsegs_single);
333 else
334 r = process_each_vg(cmd, argc, argv, LCK_VG_READ, 0,
335 report_handle, &_pvsegs_in_vg);
336 break;
337 }
338
339 dm_report_output(report_handle);
340
341 dm_report_free(report_handle);
342 return r;
343}
