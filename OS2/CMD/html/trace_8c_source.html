60/*--------------------------- global data -----------------------------*/
61static char trace_save_str[40];
62static char trace_env[40];
63static FILE *trace_fp;
64static short trace_number=0,trace_level=(-1);
65/***********************************************************************/
66#ifdef HAVE_PROTO
67void trace_initialise(void)
68#else
69void trace_initialise()
70#endif
71/***********************************************************************/
72{
73/*--------------------------- local data ------------------------------*/
74 char *trace_env_ptr=getenv("THE_TRACE");
75/*--------------------------- processing ------------------------------*/
76
77 trace_fp = NULL;
78 if (trace_env_ptr == NULL)
79 return;
80 strcpy(trace_env,trace_env_ptr);
81 trace_number = trace_level = 0;
82 strcpy(trace_save_str,"");
83 return;
84}
85/***********************************************************************/
86#ifdef HAVE_PROTO
87void trace_function(char *trace_str)
88#else
89void trace_function(trace_str)
90char *trace_str;
91#endif
92/***********************************************************************/
93{
94/*--------------------------- local data ------------------------------*/
95 register int i;
96 time_t timer;
97 struct tm *tblock=NULL;
98/*--------------------------- processing ------------------------------*/
99 if (trace_level == (-1))
100 return;
101 trace_level++;
102 trace_fp = fopen(trace_env,"a");
103 for (i=0;i<trace_level;i++)
104 fprintf(trace_fp," ");
105 fprintf(trace_fp,"(%d)%-s",trace_level,trace_str);
106 for (i=0;i<100-(trace_level*2)-strlen(trace_str);i++)
107 fprintf(trace_fp," ");
108 timer = time(NULL);
109 tblock = localtime(&timer);
110 fprintf(trace_fp,"%2d:%2d:%2d\n",tblock->tm_hour,tblock->tm_min,tblock->tm_sec);
111 fclose(trace_fp);
112 return;
113}
114/***********************************************************************/
115#ifdef HAVE_PROTO
116void trace_return(void)
117#else
118void trace_return()
119#endif
120/***********************************************************************/
121{
122/*--------------------------- local data ------------------------------*/
123/*--------------------------- processing ------------------------------*/
124 if (trace_level == (-1))
125 return;
126 trace_level--;
127 if (trace_level < 0)
128 fprintf(trace_fp,"****** trace level below zero ********");
129
130 return;
131}
132/***********************************************************************/
133#ifdef HAVE_PROTO
134void trace_string(char *fmt,...)
135#else
136void trace_string(fmt)
137char *fmt;
138#endif
139/***********************************************************************/
140{
141/*--------------------------- local data ------------------------------*/
142 va_list args;
143/*--------------------------- processing ------------------------------*/
144 va_start(args,fmt);
145 trace_fp = fopen(trace_env,"a");
146 if (trace_fp == NULL)
147 {
148 va_end(args);
149 return;
150 }
151 vfprintf(trace_fp,fmt,args);
152 fclose(trace_fp);
153 va_end(args);
154 return;
155}
156/***********************************************************************/
157#ifdef HAVE_PROTO
158void trace_constant(char *str)
159#else
160void trace_constant(str)
161char *str;
162#endif
163/***********************************************************************/
164{
165/*--------------------------- local data ------------------------------*/
166/*--------------------------- processing ------------------------------*/
167 trace_fp = fopen(trace_env,"a");
168 if (trace_fp == NULL)
169 return;
170 fprintf(trace_fp,"%s",str);
171 fclose(trace_fp);
172 return;
173}
