57int read_tags(struct dm_pool *mem, struct list *tags, struct config_value *cv)
58{
59 if (cv->type == CFG_EMPTY_ARRAY)
60 return 1;
61
62 while (cv) {
63 if (cv->type != CFG_STRING) {
64 log_error("Found a tag that is not a string");
65 return 0;
66 }
67
68 if (!str_list_add(mem, tags, dm_pool_strdup(mem, cv->v.str))) {
69 stack;
70 return 0;
71 }
72
73 cv = cv->next;
74 }

76 return 1;
