32/*
33 * Structures which define what the master boot record looks like.
34 * They are used to find the partition on disk.
35 */
36struct part {
37 UCHAR bootind; /* 0x80 means partition is bootable */
38 UCHAR starthead; /* head number of partition start */
39 UCHAR startsect; /* sector number */
40 UCHAR startcyl; /* cylinder number */
41 UCHAR systind; /* partition ID */
42 UCHAR endhead; /* head number of partition end */
43 UCHAR endsect; /* sector number */
44 UCHAR endcyl; /* cylinder number */
45 ULONG lsn; /* number of sectors before partition */
46 ULONG nsects; /* number of sectors in partition */
47};

53 struct part ptbl[4]; /* the partition table */

55};
