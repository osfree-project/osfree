41static int _create_single_area(struct dm_pool *mem, struct pv_map *pvm,
42 uint32_t start, uint32_t length)
43{
44 struct pv_area *pva;
45
46 if (!(pva = dm_pool_zalloc(mem, sizeof(*pva)))) {
47 stack;
48 return 0;
49 }
50
51 log_debug("Allowing allocation on %s start PE %" PRIu32 " length %"
52 PRIu32, dev_name(pvm->pv->dev), start, length);
53 pva->map = pvm;
54 pva->start = start;
55 pva->count = length;
56 _insert_area(&pvm->areas, pva);
57
58 return 1;
59}

61static int _create_alloc_areas_for_pv(struct dm_pool *mem, struct pv_map *pvm,
62 uint32_t start, uint32_t count)
63{
64 struct pv_segment *peg;
65 uint32_t pe, end, area_len;
66
67 /* Only select extents from start to end inclusive */
68 end = start + count - 1;
69 if (end > pvm->pv->pe_count - 1)
70 end = pvm->pv->pe_count - 1;
71
72 pe = start;
73
74 /* Walk through complete ordered list of device segments */
75 //list_iterate_items(peg, &pvm->pv->segments) {
76 list_iterate_items(peg, struct pv_segment, &pvm->pv->segments) {
77 /* pe holds the next extent we want to check */
78
79 /* Beyond the range we're interested in? */
80 if (pe > end)
81 break;
82
83 /* Skip if we haven't reached the first seg we want yet */
84 if (pe > peg->pe + peg->len - 1)
85 continue;
86
87 /* Free? */
88 if (peg->lvseg)
89 goto next;
90
91 /* How much of this peg do we need? */
92 area_len = (end >= peg->pe + peg->len - 1) ?
93 peg->len - (pe - peg->pe) : end - pe + 1;
94
95 if (!_create_single_area(mem, pvm, pe, area_len)) {
96 stack;
97 return 0;
98 }
99
100 next:
101 pe = peg->pe + peg->len;
102 }
103
104 return 1;
105}

135static int _create_maps(struct dm_pool *mem, struct list *pvs, struct list *pvms)
136{
137 struct pv_map *pvm, *pvm2;
138 struct pv_list *pvl;
139
140 //list_iterate_items(pvl, pvs) {
141 list_iterate_items(pvl, struct pv_list, pvs) {
142 if (!(pvl->pv->status & ALLOCATABLE_PV))
143 continue;
144
145 pvm = NULL;
146
147 //list_iterate_items(pvm2, pvms)
148 list_iterate_items(pvm2, struct pv_map, pvms)
149 if (pvm2->pv->dev == pvl->pv->dev) {
150 pvm = pvm2;
151 break;
152 }
153
154 if (!pvm) {
155 if (!(pvm = dm_pool_zalloc(mem, sizeof(*pvm)))) {
156 stack;
157 return 0;
158 }
159
160 pvm->pv = pvl->pv;
161 list_init(&pvm->areas);
162 list_add(pvms, &pvm->list);
163 }
164
165 if (!_create_all_areas_for_pv(mem, pvm, pvl->pe_ranges)) {
166 stack;
167 return 0;
168 }
169 }
170
171 return 1;

174/*
175 * Create list of PV areas available for this particular allocation
176 */
177struct list *create_pv_maps(struct dm_pool *mem, struct volume_group *vg,
178 struct list *allocatable_pvs)
179{
180 struct list *pvms;
181
182 if (!(pvms = dm_pool_zalloc(mem, sizeof(*pvms)))) {
183 log_error("create_pv_maps alloc failed");
184 return NULL;
185 }
186
187 list_init(pvms);
188
189 if (!_create_maps(mem, allocatable_pvs, pvms)) {
190 log_error("Couldn't create physical volume maps in %s",
191 vg->name);
192 dm_pool_free(mem, pvms);
193 return NULL;
194 }
195
196 return pvms;
197}

199void consume_pv_area(struct pv_area *pva, uint32_t to_go)
200{
201 list_del(&pva->list);
202 pva->map->pe_count -= pva->count;
203
204 assert(to_go <= pva->count);
205
206 if (to_go < pva->count) {
207 /* split the area */
208 pva->start += to_go;
209 pva->count -= to_go;
210 _insert_area(&pva->map->areas, pva);
211 }
212}
