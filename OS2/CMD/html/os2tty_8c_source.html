16// get a keystroke from STDIN, with optional echoing & CR/LF
17unsigned int GetKeystroke( int eflag )
18{
19 unsigned int c;
20 KBDKEYINFO keyinfo;
21
22 // if console redirected, we can't use KbdCharIn()
23 if (( QueryIsConsole( STDIN ) == 0 ) && (( eflag & EDIT_BIOS_KEY ) == 0 ))
24 c = getchar();
25
26 else {
27 // C Set/2 has a bug with ^C and getche()
28 // wait for a key, and check for extended scan code
29 do {
30 (void)KbdCharIn( &keyinfo, (USHORT)0, (HKBD)0 );
31 c = (((keyinfo.fbStatus & 2) && ((keyinfo.chChar == 0) || (keyinfo.chChar == 0xE0))) ? (keyinfo.chScan | 0x100) : keyinfo.chChar);
32 } while ( c == 0 );
33 }
34
35 // check for upper case requested
36 c = ((eflag & EDIT_UC_SHIFT) ? _ctoupper(c) : c);
37
38 if ((eflag & EDIT_NO_ECHO) == 0)
39 qputc( STDOUT, (char)c );
40
41 // check for CRLF requested
42 if (eflag & EDIT_ECHO_CRLF)
43 crlf();
44
45 return c;
46}

57// return the character attribute at the current cursor position
58void GetAtt(unsigned int *normal, unsigned int *inverse)
59{
60 CHAR achCells[2];
61 USHORT cb = sizeof(achCells);
62 USHORT row = 0, column = 0;
63
64 (void)VioGetCurPos( &row, &column, 0 );
65 (void)VioReadCellStr( achCells, &cb, row, column, 0 );
66 *normal = achCells[1];
67
68 // remove blink & intensity bits, and check for black on black
69 if ((*inverse = (achCells[1] & 0x77)) == 0)
70 *inverse = 0x70;
71 else {
72 // ROR 4 to get inverse
73 *inverse = (*inverse >> 4) + ((*inverse << 4) & 0xFF);
74 }
75}

171// set the cursor shape; the defaults are in gpIniptr (CursO & CursI)
172// "c_type" defaults are:
173// 0 - overstrike
174// 1 - insert
175void SetCurSize( int c_type )
176{
177 VIOCURSORINFO viociCursor;
178
179 // if CursO or CursI == -1, don't attempt to modify cursor at all
180 if ((gpIniptr->CursO != -1) && (gpIniptr->CursI != -1)) {
181
182 // if we're defaulting to insert mode, reverse the "c_type" meaning
183 if (gpIniptr->EditMode)
184 c_type = !c_type;
185
186 // set the end & start lines for the cursor cell (as percentages)
187 viociCursor.cEnd = -100;
188 viociCursor.yStart = ((c_type == 0) ? gpIniptr->CursO : gpIniptr->CursI);
189
190 // set the attribute (-1 = hidden, anything else is normal)
191 viociCursor.attr = ((viociCursor.yStart == 0) ? -1 : 0);
192 viociCursor.yStart -= 100;
193 viociCursor.cx = 0; // default character width
194 (void)VioSetCurType( &viociCursor, 0 );
195 }

202void SetBrightBG( void )
203{
204 struct {
