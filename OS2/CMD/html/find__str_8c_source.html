38/* This function prints out all lines containing a substring. There are some
39 * conditions that may be passed to the function.
40 *
41 * RETURN: If the string was found at least once, returns 1.
42 * If the string was not found at all, returns 0.
43 */
44
45/* save some stack space: */
46 unsigned char temp_str[MAX_STR];
47 unsigned char this_line[MAX_STR+3]; /* extra space for \r\n\0 */
48
49int
50find_str (unsigned char *sz, int thefile,
51 int invert_search, int count_lines, int number_output, int ignore_case)
52{
53 int i, length;
54 long line_number = 0, total_lines = 0;
55 unsigned char *c;
56 char numbuf[16]; /* for itoa */
57
58 (void)get_line (thefile, NULL, 0); /* init buffers */
59
60 /* Convert to upper if needed */
61
62 if (ignore_case)
63 {
64 length = strlen (sz);
65 for (i = 0; i < length; i++)
66 sz[i] = toupper (sz[i]);
67 }
68
69 /* Scan the file until EOF */
70
71
72 while ( get_line (thefile, temp_str, sizeof(temp_str)) )
73 /* ( fgets (temp_str, MAX_STR, p) != NULL ) */
74 /* get_line of kitten is smaller than f-stuff overhead */
75 {
76
77 /* Remove the trailing newline, if any */
78
79
80 length = strlen (temp_str);
81 if (temp_str[length-1] == '\n')
82 {
83 temp_str[length-1] = '\0';
84 }
85
86 if (temp_str[length-1] == '\r')
87 {
88 temp_str[length-1] = '\0';
89 }
90
91 /* Increment number of lines */
92
93 line_number++;
94 strcpy (this_line, temp_str);
95 /* (remembered unchanged version for showing) */
96
97 /* Convert to upper if needed */
98
99 if (ignore_case)
100 {
101 for (i = 0; i < length; i++)
102 {
103 temp_str[i] = toupper (temp_str[i]);
104 }
105 }
106
107 /* Locate the substring */
108
109 /* strstr() returns a pointer to the first occurrence in the
110 string of the substring */
111
112 /* This logic is not written well! -jh */
113
114 if ( (((c = strstr (temp_str, sz)) != NULL) && (!invert_search))
115 || ((c == NULL) && (invert_search)) )
116 {
117
118 if (!count_lines) /* only show hits if not counting them */
119 {
120 if (number_output) {
121 /* printf ("%ld:", line_number); */
122 ltoa(line_number, numbuf, 10);
123 fprintf(stderr,"%s:", numbuf);
124 }
125
126 /* Print the line of text, after adding \r\n\0 */
127
128 c = this_line + strlen(this_line);
129 c[0] = '\r';
130 c[1] = '\n';
131 c[2] = '\0';
132 fprintf(stderr, "%s", this_line);
133 }
134
135 total_lines++;
136 } /* long if */
137 } /* while fgets */
138
139 if (count_lines)
140 {
141 /* Just show num. lines that contain the string */
142 /* printf ("%ld\n", total_lines); */
143 ltoa(total_lines, numbuf, 10);
144 fprintf(stderr, "%s\r\n", numbuf);
145 }
146
147
148 /* RETURN: If the string was found at least once, returns 1.
149 * If the string was not found at all, returns 0.
150 */
151
152 return (total_lines > 0 ? 1 : 0);
153}
