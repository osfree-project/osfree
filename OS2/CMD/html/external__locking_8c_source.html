59int init_external_locking(struct locking_type *locking, struct cmd_context *cmd)
60{
61 const char *libname;
62
63 if (_locking_lib) {
64 log_error("External locking already initialised");
65 return 1;
66 }
67
68 locking->lock_resource = _lock_resource;
69 locking->fin_locking = _fin_external_locking;
70 locking->reset_locking = _reset_external_locking;
71 locking->flags = 0;
72
73 libname = find_config_tree_str(cmd, "global/locking_library",
74 DEFAULT_LOCKING_LIB);
75
76 if (!(_locking_lib = load_shared_library(cmd, libname, "locking", 1))) {
77 stack;
78 return 0;
79 }
80
81 /* Get the functions we need */
82 if (!(_init_fn = dlsym(_locking_lib, "locking_init")) ||
83 !(_lock_fn = dlsym(_locking_lib, "lock_resource")) ||
84 !(_reset_fn = dlsym(_locking_lib, "reset_locking")) ||
85 !(_end_fn = dlsym(_locking_lib, "locking_end"))) {
86 log_error("Shared library %s does not contain locking "
87 "functions", libname);
88 dlclose(_locking_lib);
89 _locking_lib = NULL;
90 return 0;
91 }
92
93 log_verbose("Loaded external locking library %s", libname);
94 return _init_fn(2, cmd->cft, &locking->flags);
95}
