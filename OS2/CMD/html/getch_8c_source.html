353struct { long length; char *address; } logical_name;

363/*--------------------------- processing ------------------------------*/

368 status = sys$assign (&logical_name, &chan, 0, 0);

372 | IO$M_TIMED, &iosb, 0, 0, &key, 1,600, 0,0, 0, 0);
373 if (!key)
374 return (0);
375 new_key = (short)(key);
