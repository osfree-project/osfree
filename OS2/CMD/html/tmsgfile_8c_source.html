2/*
3 *@@sourcefile tmsgfile.c:
4 * replacement code for DosGetMessage for decent NLS support.
5 *
6 * This code has the following advantages over DosGetMessage:
7 *
8 * 1) No external utility is necessary to change message
9 * files. Simply edit the text, and on the next call,
10 * the file is recompiled at run-time.
11 *
12 * 2) To identify messages, any string can be used, not
13 * only numerical IDs.
14 *
15 * The .TMF file must have the following format:
16 *
17 * 1) Any message must start on the beginning of a line
18 * and look like this:
19 *
20 + <--MSGID-->: message text
21 *
22 * "MSGID" can be any string and will serve as the
23 * message identifier for tmfGetMessage.
24 * Leading spaces after "-->:" will be ignored.
25 *
26 * 2) The message text may span across several lines.
27 * If so, the line breaks are returned as plain
28 * newline (\n) characters by tmfGetMessage.
29 *
30 * 3) Comments in the file are supported if they start
31 * with a semicolon (";") at the beginning of a line.
32 * Any following text is ignored until the next
33 * message ID.
34 *
35 * This file was originally added V0.9.0. The original
36 * code was contributed by Christian Langanke, but this
37 * has been completely rewritten with V0.9.16 to use my
38 * fast string functions now. Also, tmfGetMessage now
39 * requires tmfOpenMessageFile to be called beforehand
40 * and keeps all messages in memory for speed.
41 *
42 * Usage: All OS/2 programs.
43 *
44 * Function prefixes:
45 * -- tmf* text message file functions
46 *
47 * Note: Version numbering in this file relates to XWorkplace version
48 * numbering.
49 *
50 *@@header "helpers\tmsgfile.h"
51 *@@added V0.9.0 [umoeller]
52 */

107 *@@ MSGENTRY:
108 * private representation of a message in a text
109 * message file. Each tree entry in

134/*
135 *@@ LoadAndCompile:
136 * loads and compiles the message file into the
137 * given TMFMSGFILE struct.
138 *
139 * This has been extracted from tmfOpenMessageFile
140 * to allow for recompiling on the fly if the
141 * message file's last-write date/time changed.
142 *
143 * Preconditions:
144 *
145 * -- pFile->pszFilename must have been set.
146 *
147 * All other fields get initialized here.
148 *
149 *@@added V0.9.18 (2002-03-24) [umoeller]
150 *@@changed V0.9.20 (2002-07-19) [umoeller]: optimized, no longer holding all msgs im mem
151 *@@changed V1.0.1 (2002-12-16) [pr]: Ctrl-Z fix
152 */
153
154APIRET LoadAndCompile(PTMFMSGFILE pTmf, // in: TMF struct to set up
155 PXFILE pFile) // in: opened XFILE for msg file
156{
157 APIRET arc;
158
159 PSZ pszContent = NULL;
160 ULONG cbRead;
161
162 if (!(arc = doshReadText(pFile,
163 &pszContent,
164 &cbRead))) // bytes read including null char
165 {
166 // file loaded:
167
168 PCSZ pStartOfFile,
169 pStartOfMarker;
170
171 ULONG ulStartMarkerLength = strlen(G_pcszStartMarker),
172 ulEndMarkerLength = strlen(G_pcszEndMarker);
173
174 XSTRING strContents;
175
176 // initialize TMFMSGFILE struct
177 treeInit(&pTmf->IDsTreeRoot, NULL);
178
179 xstrInit(&strContents,
180 cbRead); // including null byte
181 xstrcpy(&strContents,
182 pszContent,
183 cbRead - 1); // not including null byte
184
185 // go build a tree of all message IDs...
186 pStartOfFile = strContents.psz;
187
188 // find first start message marker
189 pStartOfMarker = strstr(pStartOfFile,
190 G_pcszStartMarker); // start-of-line marker
191 while ( (pStartOfMarker)
192 && (!arc)
193 )
194 {
195 // start marker found:
196 PCSZ pStartOfMsgID = pStartOfMarker + ulStartMarkerLength;
197 // search next start marker
198 PCSZ pStartOfNextMarker = strstr(pStartOfMsgID + 1,
199 G_pcszStartMarker); // "\n<--"
200 // and the end-marker
201 PCSZ pEndOfMarker = strstr(pStartOfMsgID + 1,
202 G_pcszEndMarker); // "-->:"
203
204 PMSGENTRY pNew;
205
206 // sanity checks...
207
208 if ( (pStartOfNextMarker)
209 && (pStartOfNextMarker < pEndOfMarker)
210 )
211 {
212 // next start marker before end marker:
213 // that doesn't look correct, skip this entry
214 pStartOfMarker = pStartOfNextMarker;
215 continue;
216 }
217
218 if (!pEndOfMarker)
219 // no end marker found:
220 // that's invalid too, and there can't be any
221 // message left in the file then...
222 break;
223
224 // alright, this ID looks correct now
225 if (!(pNew = NEW(MSGENTRY)))
226 arc = ERROR_NOT_ENOUGH_MEMORY;
227 else
228 {
229 // length of the ID
230 ULONG ulIDLength = pEndOfMarker - pStartOfMsgID;
231 PCSZ pStartOfText = pEndOfMarker + ulEndMarkerLength,
232 pNextComment;
233
234 ZERO(pNew);
235
236 // copy the string ID (between start and end markers)
237 xstrInit(&pNew->strID, 0);
238 xstrcpy(&pNew->strID,
239 pStartOfMsgID,
240 ulIDLength);
241 // make ulKey point to the string ID for tree sorting
242 pNew->Tree.ulKey = (ULONG)pNew->strID.psz;
243
244 // skip leading spaces
245 while (*pStartOfText == ' ')
246 pStartOfText++;
247
248 // store offset of start of text
249 pNew->ulOfsText = pStartOfText - pStartOfFile;
250
251 // check if there's a comment before the
252 // next item
253 if (pNextComment = strstr(pStartOfText, "\n;"))
254 {
255 if ( (!pStartOfNextMarker)
256 || (pNextComment < pStartOfNextMarker)
257 )
258 pNew->cbText = // offset of comment marker
259 (pNextComment - pStartOfFile)
260 - pNew->ulOfsText;
261 else
262 // we have a next marker AND
263 // the comment comes after it
264 pNew->cbText = // offset of next marker
265 (pStartOfNextMarker - pStartOfFile)
266 - pNew->ulOfsText;
267 }
268 else
269 if (pStartOfNextMarker)
270 // other markers left:
271 pNew->cbText = // offset of next marker
272 (pStartOfNextMarker - pStartOfFile)
273 - pNew->ulOfsText;
274 else
275 // this was the last message:
276 pNew->cbText = strlen(pStartOfText);
277
278 // remove trailing newlines and Ctrl-Z
279 while ( (pNew->cbText)
280 && ( (pStartOfText[pNew->cbText - 1] == '\n')
281 || (pStartOfText[pNew->cbText - 1] == '\r')
282 || (pStartOfText[pNew->cbText - 1] == '\x1a') // V1.0.1 (2002-12-16) [pr]
283 )
284 )
285 (pNew->cbText)--;
286
287 // store this thing
288 if (!treeInsert(&pTmf->IDsTreeRoot,
289 NULL,
290 (TREE*)pNew,
291 treeCompareStrings))
292 // successfully inserted:
293 (pTmf->cIDs)++;
294 }
295
296 // go on with next start marker (can be NULL)
297 pStartOfMarker = pStartOfNextMarker;
298 } // end while ( (pStartOfMarker) ...
299
300 free(pszContent);
301
302 } // end else if (!(pTmf = NEW(TMFMSGFILE)))
303
304 return arc;
305}

307/*
308 *@@ tmfOpenMessageFile:
309 * opens a .TMF message file for future use
310 * with tmfGetMessage.
311 *
312 * Use tmfCloseMessageFile to close the file
313 * again and free all resources.
314 *
315 * Returns:
316 *
317 * -- NO_ERROR: *ppMsgFile has received the
318 * new TMFMSGFILE structure.
319 *
320 * -- ERROR_NOT_ENOUGH_MEMORY
321 *
322 * plus any of the errors of doshLoadTextFile,
323 * such as ERROR_FILE_NOT_FOUND.
324 *
325 *@@added V0.9.16 (2001-10-08) [umoeller]
326 *@@changed V0.9.20 (2002-07-19) [umoeller]: optimized, no longer holding all msgs im mem
327 */
328
329APIRET tmfOpenMessageFile(const char *pcszMessageFile, // in: fully q'fied .TMF file name
330 PTMFMSGFILE *ppMsgFile) // out: TMFMSGFILE struct
331{
332 APIRET arc;
333
334 ULONG cbFile = 0; // V1.0.7 (2005-05-21) [pr]: Workaround for kernel bug
335 PXFILE pFile;
336 if (!(arc = doshOpen(pcszMessageFile,
337 XOPEN_READ_EXISTING,
338 &cbFile,
339 &pFile)))
340 {
341 // create a TMFMSGFILE entry
342 PTMFMSGFILE pTmf;
343 if (!(pTmf = NEW(TMFMSGFILE)))
344 arc = ERROR_NOT_ENOUGH_MEMORY;
345 else
346 {
347 ZERO(pTmf);
348 pTmf->pszFilename = strdup(pcszMessageFile);
349
350 // TMFMSGFILE created:
351 if (!(arc = LoadAndCompile(pTmf, pFile)))
352 {
353 // set timestamp to that of the file
354 FILESTATUS3 fs3;
355 if (!(arc = DosQueryFileInfo(pFile->hf,
356 FIL_STANDARD,
357 &fs3,
358 sizeof(fs3))))
359 {
360 dtCreateFileTimeStamp(pTmf->szTimestamp,
361 &fs3.fdateLastWrite,
362 &fs3.ftimeLastWrite);
363
364 // output
365 *ppMsgFile = pTmf;
366 }
367 }
368
369 if (arc)
370 // error:
371 tmfCloseMessageFile(&pTmf);
372 }
373
374 doshClose(&pFile);
375 }
376
377 return arc;
378}

380/*
381 *@@ FreeInternalMem:
382 * cleans out the internal message file compilation
383 * and the content string. Used by both tmfCloseMessageFile
384 * and tmfGetMessage to allow for recompiles when the
385 * last-write date/time changed.
386 *
387 *@@added V0.9.18 (2002-03-24) [umoeller]
388 */
389
390STATIC VOID FreeInternalMem(PTMFMSGFILE pTmf)
391{
392 LONG cItems;
393 TREE** papNodes;
394
395 if ( (cItems = pTmf->cIDs)
396 && (papNodes = treeBuildArray(pTmf->IDsTreeRoot,
397 &cItems))
398 )
399 {
400 ULONG ul;
401 for (ul = 0; ul < cItems; ul++)
402 {
403 PMSGENTRY pNodeThis = (PMSGENTRY)(papNodes[ul]);
404
405 xstrClear(&pNodeThis->strID);
406
407 free(pNodeThis);
408 }
409
410 free(papNodes);
411 }
412}

416 * closes a message file opened by

419 *

423APIRET tmfCloseMessageFile(PTMFMSGFILE *ppMsgFile)
424{
425 if (ppMsgFile && *ppMsgFile)
426 {
427 PTMFMSGFILE pTmf = *ppMsgFile;
428
429 if (pTmf->pszFilename)
430 free(pTmf->pszFilename);
431
432 FreeInternalMem(pTmf);
433
434 free(pTmf);
435 *ppMsgFile = NULL;
436
437 return NO_ERROR;
438 }
439
440 return ERROR_INVALID_PARAMETER;

443/*
444 *@@ tmfGetMessage:
445 * replacement for DosGetMessage.
446 *
447 * After you have opened a .TMF file with tmfOpenMessageFile,
448 * you can pass it to this function to retrieve a message
449 * with the given string (!) ID. See tmsgfile.c for details.
450 *
451 * Note that this will invoke xstrcpy on the given XSTRING
452 * buffer. In other words, the string must be initialized
453 * (see xstrInit), but will be replaced.
454 *
455 * This does perform the same brain-dead string replacements
456 * as DosGetMessage, that is, the string "%1" will be
457 * replaced with pTable[0], "%2" will be replaced with
458 * pTable[1], and so on.
459 *
460 * Returns:
461 *
462 * -- NO_ERROR;
463 *
464 * -- ERROR_INVALID_PARAMETER
465 *
466 * -- ERROR_MR_MID_NOT_FOUND
467 *
468 *@@added V0.9.16 (2001-10-08) [umoeller]
469 *@@changed V0.9.18 (2002-03-24) [umoeller]: now recompiling if last write date changed
470 *@@changed V0.9.20 (2002-07-19) [umoeller]: optimized, no longer holding all msgs im mem
471 */
472
473APIRET tmfGetMessage(PTMFMSGFILE pMsgFile, // in: msg file opened by tmfOpenMessageFile
474 PCSZ pcszMessageName, // in: msg name to look for (case-sensitive!)
475 PXSTRING pstr, // out: message string, if found (XSTRING must be initialized)
476 PCSZ *pTable, // in: replacement table or NULL
477 ULONG cTableEntries) // in: count of items in pTable or null
478{
479 APIRET arc;
480 ULONG cbFile = 0; // V1.0.7 (2005-05-21) [pr]: Workaround for kernel bug
481 PXFILE pFile;
482
483 if ( (!pMsgFile)
484 || (!pMsgFile->pszFilename)
485 )
486 return ERROR_INVALID_PARAMETER;
487
488 // open the file again V0.9.20 (2002-07-19) [umoeller]
489 if (!(arc = doshOpen(pMsgFile->pszFilename,
490 XOPEN_READ_EXISTING,
491 &cbFile,
492 &pFile)))
493 {
494 // check if last-write date/time changed compared
495 // to the last time we opened the thing...
496 // V0.9.18 (2002-03-24) [umoeller]
497 FILESTATUS3 fs3;
498 if (!(arc = DosQueryFileInfo(pFile->hf,
499 FIL_STANDARD,
500 &fs3,
501 sizeof(fs3))))
502 {
503 CHAR szTemp[30];
504 dtCreateFileTimeStamp(szTemp,
505 &fs3.fdateLastWrite,
506 &fs3.ftimeLastWrite);
507 if (strcmp(szTemp, pMsgFile->szTimestamp))
508 {
509 // last write date changed:
510 _Pmpf((__FUNCTION__ ": timestamp changed, recompiling"));
511 FreeInternalMem(pMsgFile);
512
513 if (!(arc = LoadAndCompile(pMsgFile, pFile)))
514 strcpy(pMsgFile->szTimestamp, szTemp);
515 }
516 }
517
518 if (!arc)
519 {
520 // go find the message in the tree
521 PMSGENTRY pEntry;
522 if (!(pEntry = (PMSGENTRY)treeFind(pMsgFile->IDsTreeRoot,
523 (ULONG)pcszMessageName,
524 treeCompareStrings)))
525 arc = ERROR_MR_MID_NOT_FOUND;
526 else
527 {
528 PSZ pszMsg;
529 ULONG cbRead = pEntry->cbText;
530
531 if (!(pszMsg = (PSZ)malloc(cbRead + 1)))
532 arc = ERROR_NOT_ENOUGH_MEMORY;
533 else if (!(arc = doshReadAt(pFile,
534 pEntry->ulOfsText,
535 &cbRead,
536 pszMsg,
537 DRFL_NOCACHE | DRFL_FAILIFLESS)))
538 {
539 // null-terminate
540 pszMsg[cbRead] = '\0';
541 xstrset2(pstr,
542 pszMsg,
543 cbRead);
544
545 // kick out \r\n
546 xstrConvertLineFormat(pstr,
547 CRLF2LF);
548
549 // now replace strings from the table
550 if (cTableEntries && pTable)
551 {
552 CHAR szFind[10] = "%0";
553 ULONG ul;
554 for (ul = 0;
555 ul < cTableEntries;
556 ul++)
557 {
558 ULONG ulOfs = 0;
559
560 _ultoa(ul + 1, szFind + 1, 10);
561 while (xstrFindReplaceC(pstr,
562 &ulOfs,
563 szFind,
564 pTable[ul]))
565 ;
566 }
567 }
568 }
569 }
570 }
571
572 doshClose(&pFile);
573 }
574
575 return arc;
576}
