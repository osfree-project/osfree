53static int _get_mirror_event(char *params)
54{
55 int i, r = ME_INSYNC;
56 char **args = NULL;
57 char *dev_status_str;
58 char *log_status_str;
59 char *sync_str;
60 char *p = NULL;
61 int log_argc, num_devs;
62
63 /*
64 * dm core parms: 0 409600 mirror
65 * Mirror core parms: 2 253:4 253:5 400/400
66 * New-style failure params: 1 AA
67 * New-style log params: 3 cluster 253:3 A
68 * or 3 disk 253:3 A
69 * or 1 core
70 */
71
72 /* number of devices */
73 if (!dm_split_words(params, 1, 0, &p))
74 goto out_parse;
75
76 if (!(num_devs = atoi(p)))
77 goto out_parse;
78 p += strlen(p) + 1;
79
80 /* devices names + "400/400" + "1 AA" + 1 or 3 log parms + NULL */
81 args = dm_malloc((num_devs + 7) * sizeof(char *));
82 if (!args || dm_split_words(p, num_devs + 7, 0, args) < num_devs + 5)
83 goto out_parse;
84
85 dev_status_str = args[2 + num_devs];
86 log_argc = atoi(args[3 + num_devs]);
87 log_status_str = args[3 + num_devs + log_argc];
88 sync_str = args[num_devs];
89
90 /* Check for bad mirror devices */
91 for (i = 0; i < num_devs; i++)
92 if (dev_status_str[i] == 'D') {
93 syslog(LOG_ERR, "Mirror device, %s, has failed.\n", args[i]);
94 r = ME_FAILURE;
95 }
96
97 /* Check for bad disk log device */
98 if (log_argc > 1 && log_status_str[0] == 'D') {
99 syslog(LOG_ERR, "Log device, %s, has failed.\n",
100 args[2 + num_devs + log_argc]);
101 r = ME_FAILURE;
102 }
103
104 if (r == ME_FAILURE)
105 goto out;
106
107 p = strstr(sync_str, "/");
108 if (p) {
109 p[0] = '\0';
110 if (strcmp(sync_str, p+1))
111 r = ME_IGNORE;
112 p[0] = '/';
113 } else
114 goto out_parse;
115
116out:
117 if (args)
118 dm_free(args);
119 return r;
120
121out_parse:
122 if (args)
123 dm_free(args);
124 syslog(LOG_ERR, "Unable to parse mirror status string.");
125 return ME_IGNORE;
126}

167void process_event(struct dm_task *dmt, enum dm_event_mask event,
168 void **unused __attribute((unused)))
169{
170 void *next = NULL;
171 uint64_t start, length;
172 char *target_type = NULL;
173 char *params;
174 const char *device = dm_task_get_name(dmt);
175
176 if (pthread_mutex_trylock(&_event_mutex)) {
177 syslog(LOG_NOTICE, "Another thread is handling an event. Waiting...");
178 pthread_mutex_lock(&_event_mutex);
179 }
180 do {
181 next = dm_get_next_target(dmt, next, &start, &length,
182 &target_type, &params);
183
184 if (!target_type) {
185 syslog(LOG_INFO, "%s mapping lost.\n", device);
186 continue;
187 }
188
189 if (strcmp(target_type, "mirror")) {
190 syslog(LOG_INFO, "%s has unmirrored portion.\n", device);
191 continue;
192 }
193
194 switch(_get_mirror_event(params)) {
195 case ME_INSYNC:
196 /* FIXME: all we really know is that this
197 _part_ of the device is in sync
198 Also, this is not an error
199 */
200 syslog(LOG_NOTICE, "%s is now in-sync\n", device);
201 break;
202 case ME_FAILURE:
203 syslog(LOG_ERR, "Device failure in %s\n", device);
204 if (_remove_failed_devices(device))
205 /* FIXME Why are all the error return codes unused? Get rid of them? */
206 syslog(LOG_ERR, "Failed to remove faulty devices in %s\n",
207 device);
208 /* Should check before warning user that device is now linear
209 else
210 syslog(LOG_NOTICE, "%s is now a linear device.\n",
211 device);
212 */
213 break;
214 case ME_IGNORE:
215 break;
216 default:
217 /* FIXME Provide value then! */
218 syslog(LOG_INFO, "Unknown event received.\n");
219 }
220 } while (next);
221
222 pthread_mutex_unlock(&_event_mutex);
223}

225int register_device(const char *device, const char *uuid, int major, int minor,
226 void **unused __attribute((unused)))
227{
228 int r = 0;
229
230 pthread_mutex_lock(&_register_mutex);
231
232 syslog(LOG_INFO, "Monitoring mirror device %s for events\n", device);
233
234 /*
235 * Need some space for allocations. 1024 should be more
236 * than enough for what we need (device mapper name splitting)
237 */
238 if (!_mem_pool && !(_mem_pool = dm_pool_create("mirror_dso", 1024)))
239 goto out;
240
241 if (!_lvm_handle) {
242 lvm2_log_fn(_temporary_log_fn);
243 if (!(_lvm_handle = lvm2_init())) {
244 dm_pool_destroy(_mem_pool);
245 _mem_pool = NULL;
246 goto out;
247 }
248 lvm2_log_level(_lvm_handle, LVM2_LOG_SUPPRESS);
249 /* FIXME Temporary: move to dmeventd core */
250 lvm2_run(_lvm_handle, "_memlock_inc");
251 }
252
253 _register_count++;
254 r = 1;
255
256out:
257 pthread_mutex_unlock(&_register_mutex);
258
259 return r;
260}

262int unregister_device(const char *device, const char *uuid, int major, int minor,
263 void **unused __attribute((unused)))
264{
265 pthread_mutex_lock(&_register_mutex);
266
267 syslog(LOG_INFO, "No longer monitoring mirror device %s for events\n",
268 device);
269
270 if (!--_register_count) {
271 dm_pool_destroy(_mem_pool);
272 _mem_pool = NULL;
273 lvm2_run(_lvm_handle, "_memlock_dec");
274 lvm2_exit(_lvm_handle);
275 _lvm_handle = NULL;
276 }
277
278 pthread_mutex_unlock(&_register_mutex);
279
280 return 1;
281}
